وَ کَذَّبُوۡا وَ اتَّبَعُوۡۤا اَہۡوَآءَہُمۡ اَہۡوَ آءَہُمۡ ۚ وَ کُلُّ اَمۡرٍ مُّسۡتَقِرٌّ وَ لَقَدۡ جَآءَہُمۡ مِّنَ الۡاَنۡۢبَآءِ مَا فِیۡہِ مُزۡدَجَرٌ ۙ مُزۡدَجَرٌ حِکۡمَۃٌۢ بَالِغَۃٌ بٰلِغَۃٌ ۖ فَمَا تُغۡنِ النُّذُرُ ۙ فَتَوَلَّ عَنۡہُمۡ ۘ النُّذُرُ فَتَوَ لَّ عَنۡہُمۡ ۘ یَوۡمَ یَدۡعُ الدَّاعِ اِلٰی شَیۡءٍ نُّکُرٍ ۙ نُّکُرٍ خُشَّعًا اَبۡصَارُہُمۡ اَبۡصٰرُہُمۡ یَخۡرُجُوۡنَ مِنَ الۡاَجۡدَاثِ کَاَنَّہُمۡ جَرَادٌ مُّنۡتَشِرٌ مُّنۡتَشِرٌ ۙ مُّہۡطِعِیۡنَ اِلَی الدَّاعِ ؕ الدَّاعِ ۖ یَقُوۡلُ الۡکٰفِرُوۡنَ ہٰذَا یَوۡمٌ عَسِرٌ ۞ کَذَّبَتۡ قَبۡلَہُمۡ قَوۡمُ نُوۡحٍ فَکَذَّبُوۡا عَبۡدَنَا وَ قَالُوۡا مَجۡنُوۡنٌ وَّ ازۡدُجِرَ وَّازۡدُجِرَ فَدَعَا رَبَّہٗۤ اَنِّیۡ مَغۡلُوۡبٌ فَانۡتَصِرۡ فَفَتَحۡنَاۤ اَبۡوَابَ اَبۡوٰبَ السَّمَآءِ بِمَآءٍ مُّنۡہَمِرٍ وَّ وَ فَجَّرۡنَا الۡاَرۡضَ عُیُوۡنًا فَالۡتَقَی الۡمَآءُ عَلٰۤی اَمۡرٍ قَدۡ قُدِرَ وَ حَمَلۡنٰہُ عَلٰی ذَاتِ اَلۡوَاحٍ اَلۡوٰحٍ وَّ دُسُرٍ وَّدُسُرٍ تَجۡرِیۡ بِاَعۡیُنِنَا ۚ بِاَعۡیُنِنَا جَزَآءً لِّمَنۡ کَانَ کُفِرَ وَ لَقَدۡ لَقَد تَّرَکۡنٰہَاۤ اٰیَۃً فَہَلۡ مِنۡ مُّدَّکِرٍ فَکَیۡفَ کَانَ عَذَابِیۡ وَ نُذُرِ وَ لَقَدۡ یَسَّرۡنَا الۡقُرۡاٰنَ لِلذِّکۡرِ فَہَلۡ مِنۡ مُّدَّکِرٍ کَذَّبَتۡ عَادٌ فَکَیۡفَ کَانَ عَذَابِیۡ وَ نُذُرِ اِنَّاۤ اَرۡسَلۡنَا عَلَیۡہِمۡ رِیۡحًا صَرۡصَرًا فِیۡ یَوۡمِ نَحۡسٍ مُّسۡتَمِرٍّ تَنۡزِعُ النَّاسَ ۙ النَّاسَ کَاَنَّہُمۡ اَعۡجَازُ نَخۡلٍ مُّنۡقَعِرٍ فَکَیۡفَ کَانَ عَذَابِیۡ وَ نُذُرِ وَ لَقَدۡ یَسَّرۡنَا الۡقُرۡاٰنَ لِلذِّکۡرِ فَہَلۡ مِنۡ مُّدَّکِرٍ کَذَّبَتۡ ثَمُوۡدُ بِالنُّذُرِ فَقَالُوۡۤا اَبَشَرًا مِّنَّا وَاحِدًا نَّتَّبِعُہٗۤ ۙ وٰحِدًا نَّتَّبِعُہٗۤ اِنَّاۤ اِذًا لَّفِیۡ ضَلٰلٍ وَّ وَّسُعُرٍ اَءُلۡقِیَ سُعُرٍ ءَاُلۡقِیَ الذِّکۡرُ عَلَیۡہِ مِنۡۢ بَیۡنِنَا بَلۡ ہُوَ کَذَّابٌ اَشِرٌ سَیَعۡلَمُوۡنَ غَدًا مَّنِ الۡکَذَّابُ الۡاَشِرُ اِنَّا مُرۡسِلُوا النَّاقَۃِ فِتۡنَۃً لَّہُمۡ فَارۡتَقِبۡہُمۡ وَ اصۡطَبِرۡ وَ نَبِّئۡہُمۡ اَنَّ الۡمَآءَ قِسۡمَۃٌۢ بَیۡنَہُمۡ ۚ بَیۡنَہُمۡ ۖ کُلُّ شِرۡبٍ مُّحۡتَضَرٌ فَنَادَوۡا صَاحِبَہُمۡ فَتَعَاطٰی فَعَقَرَ فَکَیۡفَ کَانَ عَذَابِیۡ وَ نُذُرِ اِنَّاۤ اَرۡسَلۡنَا عَلَیۡہِمۡ صَیۡحَۃً وَّاحِدَۃً وّٰحِدَۃً فَکَانُوۡا کَہَشِیۡمِ الۡمُحۡتَظِرِ وَ لَقَدۡ یَسَّرۡنَا الۡقُرۡاٰنَ لِلذِّکۡرِ فَہَلۡ مِنۡ مُّدَّکِرٍ کَذَّبَتۡ قَوۡمُ لُوۡطٍۭ بِالنُّذُرِ اِنَّاۤ اَرۡسَلۡنَا عَلَیۡہِمۡ حَاصِبًا اِلَّاۤ اٰلَ لُوۡطٍ ؕ نَجَّیۡنٰہُمۡ لُوۡطٍ ۖ نَّجَّیۡنٰہُمۡ بِسَحَرٍ نِّعۡمَۃً مِّنۡ عِنۡدِنَا ؕ عِنۡدِنَا ۚ کَذٰلِکَ نَجۡزِیۡ مَنۡ شَکَرَ وَ لَقَدۡ اَنۡذَرَہُمۡ بَطۡشَتَنَا فَتَمَارَوۡا بِالنُّذُرِ وَ لَقَدۡ رَاوَدُوۡہُ رٰوَ دُوۡہُ عَنۡ ضَیۡفِہٖ فَطَمَسۡنَاۤ اَعۡیُنَہُمۡ فَذُوۡقُوۡا عَذَابِیۡ وَ نُذُرِ وَ لَقَدۡ صَبَّحَہُمۡ بُکۡرَۃً عَذَابٌ مُّسۡتَقِرٌّ فَذُوۡقُوۡا عَذَابِیۡ وَ نُذُرِ وَ لَقَدۡ یَسَّرۡنَا الۡقُرۡاٰنَ لِلذِّکۡرِ فَہَلۡ مِنۡ مُّدَّکِرٍ وَ لَقَدۡ جَآءَ اٰلَ فِرۡعَوۡنَ النُّذُرُ کَذَّبُوۡا بِاٰیٰتِنَا کُلِّہَا فَاَخَذۡنٰہُمۡ اَخۡذَ عَزِیۡزٍ مُّقۡتَدِرٍ اَکُفَّارُکُمۡ خَیۡرٌ مِّنۡ اُولٰٓئِکُمۡ اَمۡ لَکُمۡ بَرَآءَۃٌ فِی الزُّبُرِ اَمۡ یَقُوۡلُوۡنَ نَحۡنُ جَمِیۡعٌ مُّنۡتَصِرٌ سَیُہۡزَمُ الۡجَمۡعُ وَ یُوَ لُّوۡنَ یُوَلُّوۡنَ الدُّبُرَ بَلِ السَّاعَۃُ مَوۡعِدُہُمۡ وَ السَّاعَۃُ اَدۡہٰی وَ اَمَرُّ اِنَّ الۡمُجۡرِمِیۡنَ فِیۡ ضَلٰلٍ وَّ سُعُرٍ وَّسُعُرٍ یَوۡمَ یُسۡحَبُوۡنَ فِی النَّارِ عَلٰی وُجُوۡہِہِمۡ ؕ وُجُوۡہِہِمۡ ذُوۡقُوۡا مَسَّ سَقَرَ اِنَّا کُلَّ شَیۡءٍ خَلَقۡنٰہُ بِقَدَرٍ وَ مَاۤ اَمۡرُنَاۤ اِلَّا وَاحِدَۃٌ وٰحِدَۃٌ کَلَمۡحٍۭ بِالۡبَصَرِ وَ لَقَدۡ اَہۡلَکۡنَاۤ اَشۡیَاعَکُمۡ فَہَلۡ مِنۡ مُّدَّکِرٍ وَ کُلُّ شَیۡءٍ فَعَلُوۡہُ فِی الزُّبُرِ وَ کُلُّ صَغِیۡرٍ وَّ کَبِیۡرٍ وَّکَبِیۡرٍ مُّسۡتَطَرٌ اِنَّ الۡمُتَّقِیۡنَ فِیۡ جَنّٰتٍ وَّ نَہَرٍ وَّنَہَرٍ فِیۡ مَقۡعَدِ صِدۡقٍ عِنۡدَ مَلِیۡکٍ مُّقۡتَدِرٍ مُّقۡتَدِرٍۭ بِسۡمِ اللّٰہِ الرَّحۡمٰنِ الرَّحِیۡمِ اَلرَّحۡمٰنُ ۙ اَلرَّحۡمٰنُ عَلَّمَ الۡقُرۡاٰنَ ؕ الۡقُرۡاٰنَ خَلَقَ الۡاِنۡسَانَ ۙ الۡاِنۡسٰنَ عَلَّمَہُ الۡبَیَانَ اَلشَّمۡسُ وَ الۡقَمَرُ بِحُسۡبَانٍ وَّ وَ النَّجۡمُ وَ الشَّجَرُ یَسۡجُدٰنِ یَسۡجُدَانِ وَ السَّمَآءَ رَفَعَہَا وَ وَضَعَ وَ ضَعَ الۡمِیۡزَانَ الۡمِیۡزَانَ ۙ اَلَّا تَطۡغَوۡا فِی الۡمِیۡزَانِ وَ اَقِیۡمُوا الۡوَزۡنَ الۡوَ زۡنَ بِالۡقِسۡطِ وَ لَا تُخۡسِرُوا الۡمِیۡزَانَ وَ الۡاَرۡضَ وَضَعَہَا وَ ضَعَہَا لِلۡاَنَامِ فِیۡہَا فَاکِہَۃٌ ۪ۙ فٰکِہَۃٌ وَّ النَّخۡلُ وَّالنَّخۡلُ ذَاتُ الۡاَکۡمَامِ وَ الۡحَبُّ ذُو الۡعَصۡفِ وَ الرَّیۡحَانُ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ خَلَقَ الۡاِنۡسَانَ الۡاِنۡسٰنَ مِنۡ صَلۡصَالٍ صَلۡصٰلٍ کَالۡفَخَّارِ وَ خَلَقَ الۡجَآنَّ مِنۡ مَّارِجٍ مِّنۡ نَّارٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ رَبُّ الۡمَشۡرِقَیۡنِ وَ رَبُّ الۡمَغۡرِبَیۡنِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ مَرَجَ الۡبَحۡرَیۡنِ یَلۡتَقِیٰنِ یَلۡتَقِیَانِ بَیۡنَہُمَا بَرۡزَخٌ لَّا یَبۡغِیٰنِ یَبۡغِیَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ یَخۡرُجُ مِنۡہُمَا اللُّؤۡلُؤُ وَ الۡمَرۡجَانُ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ وَ لَہُ الۡجَوَارِ الۡجَوَ ارِ الۡمُنۡشَاٰتُ الۡمُنۡشَئٰتُ فِی الۡبَحۡرِ کَالۡاَعۡلَامِ کَالۡاَعۡلٰمِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ کُلُّ مَنۡ عَلَیۡہَا فَانٍ وَّ وَ یَبۡقٰی وَجۡہُ وَ جۡہُ رَبِّکَ ذُو الۡجَلٰلِ وَ الۡاِکۡرَامِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ یَسۡـَٔلُہٗ مَنۡ فِی السَّمٰوٰتِ وَ الۡاَرۡضِ ؕ الۡاَرۡضِ ۚ کُلَّ یَوۡمٍ ہُوَ فِیۡ شَاۡنٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ سَنَفۡرُغُ لَکُمۡ اَیُّہَ الثَّقَلٰنِ الثَّقَلَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ یٰمَعۡشَرَ الۡجِنِّ وَ الۡاِنۡسِ اِنِ اسۡتَطَعۡتُمۡ اَنۡ تَنۡفُذُوۡا مِنۡ اَقۡطَارِ السَّمٰوٰتِ وَ الۡاَرۡضِ فَانۡفُذُوۡا ؕ فَانۡفُذُوۡا ۚ لَا تَنۡفُذُوۡنَ اِلَّا بِسُلۡطٰنٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ یُرۡسَلُ عَلَیۡکُمَا شُوَاظٌ شُوَ اظٌ مِّنۡ نَّارٍ ۬ۙ وَّ نَّارٍ وَّنُحَاسٌ نُحَاسٌ فَلَا تَنۡتَصِرٰنِ تَنۡتَصِرَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فَاِذَا انۡشَقَّتِ السَّمَآءُ فَکَانَتۡ وَرۡدَۃً وَ رۡدَۃً کَالدِّہَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فَیَوۡمَئِذٍ لَّا یُسۡـَٔلُ عَنۡ ذَنۡۢبِہٖۤ اِنۡسٌ وَّ لَا وَّلَا جَآنٌّ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ یُعۡرَفُ الۡمُجۡرِمُوۡنَ بِسِیۡمٰہُمۡ فَیُؤۡخَذُ بِالنَّوَاصِیۡ بِالنَّوٰصِیۡ وَ الۡاَقۡدَامِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ ہٰذِہٖ جَہَنَّمُ الَّتِیۡ یُکَذِّبُ بِہَا الۡمُجۡرِمُوۡنَ یَطُوۡفُوۡنَ بَیۡنَہَا وَ بَیۡنَ حَمِیۡمٍ اٰنٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ وَ لِمَنۡ خَافَ مَقَامَ رَبِّہٖ جَنَّتٰنِ جَنَّتَانِ
