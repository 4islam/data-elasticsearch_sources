ہُوَ الَّذِیۡ یُصَوِّرُکُمۡ فِی الۡاَرۡحَامِ کَیۡفَ یَشَآءُ ؕ یَشَآءُ ۚ لَاۤ اِلٰہَ اِلَّا ہُوَ الۡعَزِیۡزُ الۡحَکِیۡمُ ہُوَ الَّذِیۡۤ اَنۡزَلَ عَلَیۡکَ الۡکِتٰبَ مِنۡہُ اٰیٰتٌ مُّحۡکَمٰتٌ ہُنَّ اُمُّ الۡکِتٰبِ وَ اُخَرُ مُتَشٰبِہٰتٌ ؕ مُتَشٰبِہٰتٌ ۖ فَاَمَّا الَّذِیۡنَ فِیۡ قُلُوۡبِہِمۡ زَیۡغٌ فَیَتَّبِعُوۡنَ مَا تَشَابَہَ تَشٰبَہَ مِنۡہُ ابۡتِغَآءَ الۡفِتۡنَۃِ وَ ابۡتِغَآءَ تَاۡوِیۡلِہٖ ۚ؃ وَ تَاۡوِیۡلِہٖ ۗ وَ مَا یَعۡلَمُ تَاۡوِیۡلَہٗۤ اِلَّا اللّٰہُ ۘؔ ۗ وَ الرّٰسِخُوۡنَ فِی الۡعِلۡمِ یَقُوۡلُوۡنَ اٰمَنَّا بِہٖ ۙ بِہٖ کُلٌّ مِّنۡ عِنۡدِ رَبِّنَا ۚ رَبِّنَا ۗ وَ مَا یَذَّکَّرُ اِلَّاۤ اُولُوا الۡاَلۡبَابِ الۡاَلۡبٰبِ رَبَّنَا لَا تُزِغۡ قُلُوۡبَنَا بَعۡدَ اِذۡ ہَدَیۡتَنَا وَ ہَبۡ لَنَا مِنۡ لَّدُنۡکَ رَحۡمَۃً ۚ رَحۡمَۃً ۚ اِنَّکَ اَنۡتَ الۡوَہَّابُ الۡوَ ہَّابُ رَبَّنَاۤ اِنَّکَ جَامِعُ النَّاسِ لِیَوۡمٍ لَّا رَیۡبَ فِیۡہِ ؕ فِیۡہِ ۚ اِنَّ اللّٰہَ لَا یُخۡلِفُ الۡمِیۡعَادَ اِنَّ الَّذِیۡنَ کَفَرُوۡا لَنۡ تُغۡنِیَ عَنۡہُمۡ اَمۡوَالُہُمۡ اَمۡوٰلُہُمۡ وَ لَاۤ اَوۡلَادُہُمۡ اَوۡلٰدُہُمۡ مِّنَ اللّٰہِ شَیۡئًا ؕ وَ شَیۡـًٔا ۖ وَّاُولٰٓئِکَ اُولٰٓئِکَ ہُمۡ وَقُوۡدُ وَ قُوۡدُ النَّارِ کَدَاۡبِ اٰلِ فِرۡعَوۡنَ ۙ فِرۡعَوۡنَ وَ الَّذِیۡنَ مِنۡ قَبۡلِہِمۡ ؕ قَبۡلِہِمۡ ۚ کَذَّبُوۡا بِاٰیٰتِنَا ۚ بِاٰیٰتِنَا فَاَخَذَہُمُ اللّٰہُ بِذُنُوۡبِہِمۡ ؕ بِذُنُوۡبِہِمۡ ۗ وَ اللّٰہُ شَدِیۡدُ الۡعِقَابِ قُلۡ قُل لِّلَّذِیۡنَ کَفَرُوۡا سَتُغۡلَبُوۡنَ وَ تُحۡشَرُوۡنَ اِلٰی جَہَنَّمَ ؕ جَہَنَّمَ ۚ وَ بِئۡسَ الۡمِہَادُ قَدۡ کَانَ لَکُمۡ اٰیَۃٌ فِیۡ فِئَتَیۡنِ الۡتَقَتَا ؕ الۡتَقَتَا ۖ فِئَۃٌ تُقَاتِلُ تُقٰتِلُ فِیۡ سَبِیۡلِ اللّٰہِ وَ اُخۡرٰی کَافِرَۃٌ یَّرَوۡنَہُمۡ مِّثۡلَیۡہِمۡ رَاۡیَ الۡعَیۡنِ ؕ الۡعَیۡنِ ۚ وَ اللّٰہُ یُؤَیِّدُ بِنَصۡرِہٖ مَنۡ یَّشَآءُ ؕ یَّشَآءُ ۗ اِنَّ فِیۡ ذٰلِکَ لَعِبۡرَۃً لِّاُولِی الۡاَبۡصَارِ الۡاَبۡصٰرِ زُیِّنَ لِلنَّاسِ حُبُّ الشَّہَوٰتِ مِنَ النِّسَآءِ وَ الۡبَنِیۡنَ وَ الۡقَنَاطِیۡرِ الۡقَنٰطِیۡرِ الۡمُقَنۡطَرَۃِ مِنَ الذَّہَبِ وَ الۡفِضَّۃِ وَ الۡخَیۡلِ الۡمُسَوَّمَۃِ وَ الۡاَنۡعَامِ الۡاَنۡعٰمِ وَ الۡحَرۡثِ ؕ الۡحَرۡثِ ۗ ذٰلِکَ مَتَاعُ مَتٰعُ الۡحَیٰوۃِ الدُّنۡیَا ۚ الدُّنۡیَا ۖ وَ اللّٰہُ عِنۡدَہٗ حُسۡنُ الۡمَاٰبِ ۞ قُلۡ اَؤُنَبِّئُکُمۡ بِخَیۡرٍ مِّنۡ ذٰلِکُمۡ ؕ ذٰلِکُمۡ ۚ لِلَّذِیۡنَ اتَّقَوۡا عِنۡدَ رَبِّہِمۡ جَنّٰتٌ تَجۡرِیۡ مِنۡ تَحۡتِہَا الۡاَنۡہٰرُ خٰلِدِیۡنَ فِیۡہَا وَ اَزۡوَاجٌ اَزۡوٰجٌ مُّطَہَّرَۃٌ وَّ رِضۡوَانٌ وَّرِضۡوٰنٌ مِّنَ اللّٰہِ ؕ اللّٰہِ ۗ وَ اللّٰہُ بَصِیۡرٌۢ بِالۡعِبَادِ اَلَّذِیۡنَ یَقُوۡلُوۡنَ رَبَّنَاۤ اِنَّنَاۤ اٰمَنَّا فَاغۡفِرۡ لَنَا ذُنُوۡبَنَا وَ قِنَا عَذَابَ النَّارِ اَلصّٰبِرِیۡنَ وَ الصّٰدِقِیۡنَ وَ الۡقٰنِتِیۡنَ وَ الۡمُنۡفِقِیۡنَ وَ الۡمُسۡتَغۡفِرِیۡنَ بِالۡاَسۡحَارِ شَہِدَ اللّٰہُ اَنَّہٗ لَاۤ اِلٰہَ اِلَّا ہُوَ ۙ ہُوَ وَ الۡمَلٰٓئِکَۃُ وَ اُولُوا الۡعِلۡمِ قَآئِمًۢا بِالۡقِسۡطِ ؕ بِالۡقِسۡطِ ۚ لَاۤ اِلٰہَ اِلَّا ہُوَ الۡعَزِیۡزُ الۡحَکِیۡمُ ؕ اِنَّ الدِّیۡنَ عِنۡدَ اللّٰہِ الۡاِسۡلَامُ ۟ الۡاِسۡلٰمُ ۗ وَ مَا اخۡتَلَفَ الَّذِیۡنَ اُوۡتُوا الۡکِتٰبَ اِلَّا مِنۡۢ بَعۡدِ مَا جَآءَہُمُ الۡعِلۡمُ بَغۡیًۢا بَیۡنَہُمۡ ؕ بَیۡنَہُمۡ ۗ وَ مَنۡ یَّکۡفُرۡ بِاٰیٰتِ اللّٰہِ فَاِنَّ اللّٰہَ سَرِیۡعُ الۡحِسَابِ فَاِنۡ حَآجُّوۡکَ فَقُلۡ اَسۡلَمۡتُ وَجۡہِیَ وَ جۡہِیَ لِلّٰہِ وَ مَنِ اتَّبَعَنِ ؕ اتَّبَعَنِ ۗ وَ قُلۡ قُل لِّلَّذِیۡنَ اُوۡتُوا الۡکِتٰبَ وَ الۡاُمِّیّٖنَ ءَاَسۡلَمۡتُمۡ ؕ ءَاَسۡلَمۡتُمۡ ۚ فَاِنۡ اَسۡلَمُوۡا فَقَدِ اہۡتَدَوۡا ۚ وَ اہۡتَدَوۡا اِنۡ ۖ وَّاِنۡ تَوَ لَّوۡا تَوَلَّوۡا فَاِنَّمَا عَلَیۡکَ الۡبَلٰغُ ؕ الۡبَلٰغُ ۗ وَ اللّٰہُ بَصِیۡرٌۢ بِالۡعِبَادِ اِنَّ الَّذِیۡنَ یَکۡفُرُوۡنَ بِاٰیٰتِ اللّٰہِ وَ یَقۡتُلُوۡنَ النَّبِیّٖنَ بِغَیۡرِ حَقٍّ ۙ حَقٍّ وَّ یَقۡتُلُوۡنَ وَّیَقۡتُلُوۡنَ الَّذِیۡنَ یَاۡمُرُوۡنَ بِالۡقِسۡطِ مِنَ النَّاسِ ۙ النَّاسِ فَبَشِّرۡہُمۡ بِعَذَابٍ اَلِیۡمٍ اُولٰٓئِکَ الَّذِیۡنَ حَبِطَتۡ اَعۡمَالُہُمۡ اَعۡمٰلُہُمۡ فِی الدُّنۡیَا وَ الۡاٰخِرَۃِ ۫ الۡاٰخِرَۃِ وَ مَا لَہُمۡ مِّنۡ نّٰصِرِیۡنَ اَلَمۡ تَرَ اِلَی الَّذِیۡنَ اُوۡتُوۡا نَصِیۡبًا مِّنَ الۡکِتٰبِ یُدۡعَوۡنَ اِلٰی کِتٰبِ اللّٰہِ لِیَحۡکُمَ بَیۡنَہُمۡ ثُمَّ یَتَوَلّٰی یَتَوَ لّٰی فَرِیۡقٌ مِّنۡہُمۡ وَ ہُمۡ مُّعۡرِضُوۡنَ ذٰلِکَ بِاَنَّہُمۡ قَالُوۡا لَنۡ تَمَسَّنَا النَّارُ اِلَّاۤ اَیَّامًا مَّعۡدُوۡدٰتٍ ۪ وَ مَّعۡدُوۡدٰتٍ ۖ وَّغَرَّہُمۡ غَرَّہُمۡ فِیۡ دِیۡنِہِمۡ مَّا کَانُوۡا یَفۡتَرُوۡنَ فَکَیۡفَ اِذَا جَمَعۡنٰہُمۡ لِیَوۡمٍ لَّا رَیۡبَ فِیۡہِ ۟ فِیۡہِ وَ وُفِّیَتۡ کُلُّ نَفۡسٍ مَّا کَسَبَتۡ وَ ہُمۡ لَا یُظۡلَمُوۡنَ قُلِ اللّٰہُمَّ مٰلِکَ الۡمُلۡکِ تُؤۡتِی الۡمُلۡکَ مَنۡ تَشَآءُ وَ تَنۡزِعُ الۡمُلۡکَ مِمَّنۡ تَشَآءُ ۫ تَشَآءُ وَ تُعِزُّ مَنۡ تَشَآءُ وَ تُذِلُّ مَنۡ تَشَآءُ ؕ تَشَآءُ ۖ بِیَدِکَ الۡخَیۡرُ ؕ الۡخَیۡرُ ۖ اِنَّکَ عَلٰی کُلِّ شَیۡءٍ قَدِیۡرٌ تُوۡلِجُ الَّیۡلَ فِی النَّہَارِ وَ تُوۡلِجُ النَّہَارَ فِی الَّیۡلِ ۫ الَّیۡلِ ۖ وَ تُخۡرِجُ الۡحَیَّ مِنَ الۡمَیِّتِ وَ تُخۡرِجُ الۡمَیِّتَ مِنَ الۡحَیِّ ۫ الۡحَیِّ ۖ وَ تَرۡزُقُ مَنۡ تَشَآءُ بِغَیۡرِ حِسَابٍ لَا لَّا یَتَّخِذِ الۡمُؤۡمِنُوۡنَ الۡکٰفِرِیۡنَ اَوۡلِیَآءَ مِنۡ دُوۡنِ الۡمُؤۡمِنِیۡنَ ۚ الۡمُؤۡمِنِیۡنَ ۖ وَ مَنۡ یَّفۡعَلۡ ذٰلِکَ فَلَیۡسَ مِنَ اللّٰہِ فِیۡ شَیۡءٍ اِلَّاۤ اَنۡ تَتَّقُوۡا مِنۡہُمۡ تُقٰٮۃً ؕ تُقٰٮۃً ۗ وَ یُحَذِّرُکُمُ اللّٰہُ نَفۡسَہٗ ؕ نَفۡسَہٗ ۗ وَ اِلَی اللّٰہِ الۡمَصِیۡرُ قُلۡ اِنۡ تُخۡفُوۡا مَا فِیۡ صُدُوۡرِکُمۡ اَوۡ تُبۡدُوۡہُ یَعۡلَمۡہُ اللّٰہُ ؕ اللّٰہُ ۗ وَ یَعۡلَمُ مَا فِی السَّمٰوٰتِ وَ مَا فِی الۡاَرۡضِ ؕ الۡاَرۡضِ ۗ وَ اللّٰہُ عَلٰی کُلِّ شَیۡءٍ قَدِیۡرٌ یَوۡمَ تَجِدُ کُلُّ نَفۡسٍ مَّا عَمِلَتۡ مِنۡ خَیۡرٍ مُّحۡضَرًا ۚۖۛ وَّ مُّحۡضَرًا وَّمَا مَا عَمِلَتۡ مِنۡ سُوۡٓءٍ ۚۛ تَوَدُّ سُوۡٓءٍ تَوَ دُّ لَوۡ اَنَّ بَیۡنَہَا وَ بَیۡنَہٗۤ اَمَدًۢا بَعِیۡدًا ؕ بَعِیۡدًا ۗ وَ یُحَذِّرُکُمُ اللّٰہُ نَفۡسَہٗ ؕ نَفۡسَہٗ ۗ وَ اللّٰہُ رَءُوۡفٌۢ بِالۡعِبَادِ قُلۡ اِنۡ کُنۡتُمۡ تُحِبُّوۡنَ اللّٰہَ فَاتَّبِعُوۡنِیۡ یُحۡبِبۡکُمُ اللّٰہُ وَ یَغۡفِرۡ لَکُمۡ ذُنُوۡبَکُمۡ ؕ ذُنُوۡبَکُمۡ ۗ وَ اللّٰہُ غَفُوۡرٌ رَّحِیۡمٌ قُلۡ اَطِیۡعُوا اللّٰہَ وَ الرَّسُوۡلَ ۚ الرَّسُوۡلَ ۖ فَاِنۡ تَوَلَّوۡا تَوَ لَّوۡا فَاِنَّ اللّٰہَ لَا یُحِبُّ الۡکٰفِرِیۡنَ ۞ اِنَّ اللّٰہَ اصۡطَفٰۤی اٰدَمَ وَ نُوۡحًا وَّ اٰلَ وَّاٰلَ اِبۡرٰہِیۡمَ وَ اٰلَ عِمۡرٰنَ عَلَی الۡعٰلَمِیۡنَ ذُرِّیَّۃًۢ بَعۡضُہَا مِنۡۢ بَعۡضٍ ؕ بَعۡضٍ ۗ وَ اللّٰہُ سَمِیۡعٌ عَلِیۡمٌ اِذۡ قَالَتِ امۡرَاَتُ عِمۡرٰنَ رَبِّ اِنِّیۡ نَذَرۡتُ لَکَ مَا فِیۡ بَطۡنِیۡ مُحَرَّرًا فَتَقَبَّلۡ مِنِّیۡ ۚ مِنِّیۡۤ ۖ اِنَّکَ اَنۡتَ السَّمِیۡعُ الۡعَلِیۡمُ فَلَمَّا وَضَعَتۡہَا وَ ضَعَتۡہَا قَالَتۡ رَبِّ اِنِّیۡ وَضَعۡتُہَاۤ وَ ضَعۡتُہَاۤ اُنۡثٰی اُنۡثٰی ؕ وَ اللّٰہُ اَعۡلَمُ بِمَا وَضَعَتۡ ؕ وَ ضَعَتۡ وَ لَیۡسَ الذَّکَرُ کَالۡاُنۡثٰی ۚ کَالۡاُنۡثٰی ۖ وَ اِنِّیۡ سَمَّیۡتُہَا اَنَا۠ مَعَکُمۡ مِّنَ الشّٰہِدِیۡنَ فَمَنۡ تَوَلّٰی تَوَ لّٰی بَعۡدَ ذٰلِکَ فَاُولٰٓئِکَ ہُمُ الۡفٰسِقُوۡنَ اَفَغَیۡرَ دِیۡنِ اللّٰہِ یَبۡغُوۡنَ وَ لَہٗۤ اَسۡلَمَ مَنۡ فِی السَّمٰوٰتِ وَ الۡاَرۡضِ طَوۡعًا وَّ کَرۡہًا وَّکَرۡہًا وَّ اِلَیۡہِ وَّاِلَیۡہِ یُرۡجَعُوۡنَ قُلۡ اٰمَنَّا بِاللّٰہِ وَ مَاۤ اُنۡزِلَ عَلَیۡنَا وَ مَاۤ اُنۡزِلَ عَلٰۤی اِبۡرٰہِیۡمَ وَ اِسۡمٰعِیۡلَ وَ اِسۡحٰقَ وَ یَعۡقُوۡبَ
