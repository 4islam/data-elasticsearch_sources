وَ قٰتِلُوۡہُمۡ حَتّٰی لَا تَکُوۡنَ فِتۡنَۃٌ وَّ یَکُوۡنَ وَّیَکُوۡنَ الدِّیۡنُ لِلّٰہِ ؕ لِلّٰہِ ۖ فَاِنِ انۡتَہَوۡا فَلَا عُدۡوَانَ عُدۡوٰنَ اِلَّا عَلَی الظّٰلِمِیۡنَ اَلشَّہۡرُ الۡحَرَامُ بِالشَّہۡرِ الۡحَرَامِ وَ الۡحُرُمٰتُ قِصَاصٌ ؕ قِصَاصٌ ۚ فَمَنِ اعۡتَدٰی عَلَیۡکُمۡ فَاعۡتَدُوۡا عَلَیۡہِ بِمِثۡلِ مَا اعۡتَدٰی عَلَیۡکُمۡ ۪ ۚ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ مَعَ الۡمُتَّقِیۡنَ وَ اَنۡفِقُوۡا فِیۡ سَبِیۡلِ اللّٰہِ وَ لَا تُلۡقُوۡا بِاَیۡدِیۡکُمۡ اِلَی التَّہۡلُکَۃِ ۚۖۛ التَّہۡلُکَۃِ ۛ وَ اَحۡسِنُوۡا ۚۛ اَحۡسِنُوۡۤا ۛ اِنَّ اللّٰہَ یُحِبُّ الۡمُحۡسِنِیۡنَ وَ اَتِمُّوا الۡحَجَّ وَ الۡعُمۡرَۃَ لِلّٰہِ ؕ لِلّٰہِ ۚ فَاِنۡ اُحۡصِرۡتُمۡ فَمَا اسۡتَیۡسَرَ مِنَ الۡہَدۡیِ ۚ الۡہَدۡیِ ۖ وَ لَا تَحۡلِقُوۡا رُءُوۡسَکُمۡ حَتّٰی یَبۡلُغَ الۡہَدۡیُ مَحِلَّہٗ ؕ مَحِلَّہٗ ۚ فَمَنۡ کَانَ مِنۡکُمۡ مَّرِیۡضًا اَوۡ بِہٖۤ اَذًی مِّنۡ رَّاۡسِہٖ فَفِدۡیَۃٌ مِّنۡ صِیَامٍ اَوۡ صَدَقَۃٍ اَوۡ نُسُکٍ ۚ نُسُکٍ ۚ فَاِذَاۤ اَمِنۡتُمۡ ٝ اَمِنۡتُمۡ فَمَنۡ تَمَتَّعَ بِالۡعُمۡرَۃِ اِلَی الۡحَجِّ فَمَا اسۡتَیۡسَرَ مِنَ الۡہَدۡیِ ۚ الۡہَدۡیِ ۚ فَمَنۡ لَّمۡ یَجِدۡ فَصِیَامُ ثَلٰثَۃِ اَیَّامٍ فِی الۡحَجِّ وَ سَبۡعَۃٍ اِذَا رَجَعۡتُمۡ ؕ رَجَعۡتُمۡ ۗ تِلۡکَ عَشَرَۃٌ کَامِلَۃٌ ؕ ۗ ذٰلِکَ لِمَنۡ لَّمۡ یَکُنۡ اَہۡلُہٗ حَاضِرِی الۡمَسۡجِدِ الۡحَرَامِ ؕ الۡحَرَامِ ۚ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ شَدِیۡدُ الۡعِقَابِ اَلۡحَجُّ اَشۡہُرٌ مَّعۡلُوۡمٰتٌ ۚ مَّعۡلُوۡمٰتٌ ۚ فَمَنۡ فَرَضَ فِیۡہِنَّ الۡحَجَّ فَلَا رَفَثَ وَ لَا فُسُوۡقَ ۙ وَ لَا جِدَالَ فِی الۡحَجِّ ؕ الۡحَجِّ ۗ وَ مَا تَفۡعَلُوۡا مِنۡ خَیۡرٍ یَّعۡلَمۡہُ اللّٰہُ ؕؔ ۗ وَ تَزَوَّدُوۡا فَاِنَّ خَیۡرَ الزَّادِ التَّقۡوٰی ۫ التَّقۡوٰی ۚ وَ اتَّقُوۡنِ یٰۤاُولِی الۡاَلۡبَابِ الۡاَلۡبٰبِ لَیۡسَ عَلَیۡکُمۡ جُنَاحٌ اَنۡ تَبۡتَغُوۡا فَضۡلًا مِّنۡ رَّبِّکُمۡ ؕ رَّبِّکُمۡ ۚ فَاِذَاۤ اَفَضۡتُمۡ مِّنۡ عَرَفٰتٍ فَاذۡکُرُوا اللّٰہَ عِنۡدَ الۡمَشۡعَرِ الۡحَرَامِ ۪ الۡحَرَامِ ۖ وَ اذۡکُرُوۡہُ کَمَا ہَدٰٮکُمۡ ۚ ہَدٰٮکُمۡ وَ اِنۡ کُنۡتُمۡ مِّنۡ قَبۡلِہٖ لَمِنَ الضَّآلِّیۡنَ ثُمَّ اَفِیۡضُوۡا مِنۡ حَیۡثُ اَفَاضَ النَّاسُ وَ اسۡتَغۡفِرُوا اللّٰہَ ؕ ۚ اِنَّ اللّٰہَ غَفُوۡرٌ رَّحِیۡمٌ فَاِذَا قَضَیۡتُمۡ مَّنَاسِکَکُمۡ مَّنٰسِکَکُمۡ فَاذۡکُرُوا اللّٰہَ کَذِکۡرِکُمۡ اٰبَآءَکُمۡ اَوۡ اَشَدَّ ذِکۡرًا ؕ ذِکۡرًا ۗ فَمِنَ النَّاسِ مَنۡ یَّقُوۡلُ رَبَّنَاۤ اٰتِنَا فِی الدُّنۡیَا وَ مَا لَہٗ فِی الۡاٰخِرَۃِ مِنۡ خَلَاقٍ خَلٰقٍ وَ مِنۡہُمۡ مَّنۡ یَّقُوۡلُ رَبَّنَاۤ اٰتِنَا فِی الدُّنۡیَا حَسَنَۃً وَّ وَّفِی فِی الۡاٰخِرَۃِ حَسَنَۃً وَّ وَّقِنَا قِنَا عَذَابَ النَّارِ اُولٰٓئِکَ لَہُمۡ نَصِیۡبٌ مِّمَّا کَسَبُوۡا ؕ کَسَبُوۡا ۚ وَ اللّٰہُ سَرِیۡعُ الۡحِسَابِ ۞ وَ اذۡکُرُوا اللّٰہَ فِیۡۤ اَیَّامٍ مَّعۡدُوۡدٰتٍ ؕ مَّعۡدُوۡدٰتٍ ۚ فَمَنۡ تَعَجَّلَ فِیۡ یَوۡمَیۡنِ فَلَاۤ اِثۡمَ عَلَیۡہِ ۚ عَلَیۡہِ وَ مَنۡ تَاَخَّرَ فَلَاۤ اِثۡمَ عَلَیۡہِ ۙ ۚ لِمَنِ اتَّقٰی ؕ اتَّقٰی ۗ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّکُمۡ اِلَیۡہِ تُحۡشَرُوۡنَ وَ مِنَ النَّاسِ مَنۡ یُّعۡجِبُکَ قَوۡلُہٗ فِی الۡحَیٰوۃِ الدُّنۡیَا وَ یُشۡہِدُ اللّٰہَ عَلٰی مَا فِیۡ قَلۡبِہٖ ۙ وَ ہُوَ اَلَدُّ الۡخِصَامِ وَ اِذَا تَوَلّٰی تَوَ لّٰی سَعٰی فِی الۡاَرۡضِ لِیُفۡسِدَ فِیۡہَا وَ یُہۡلِکَ الۡحَرۡثَ وَ النَّسۡلَ ؕ النَّسۡلَ ۗ وَ اللّٰہُ لَا یُحِبُّ الۡفَسَادَ وَ اِذَا قِیۡلَ لَہُ اتَّقِ اللّٰہَ اَخَذَتۡہُ الۡعِزَّۃُ بِالۡاِثۡمِ ۚ فَحَسۡبُہٗ جَہَنَّمُ ؕ جَہَنَّمُ ۚ وَ لَبِئۡسَ الۡمِہَادُ وَ مِنَ النَّاسِ مَنۡ یَّشۡرِیۡ نَفۡسَہُ ابۡتِغَآءَ مَرۡضَاتِ اللّٰہِ ؕ اللّٰہِ ۗ وَ اللّٰہُ رَءُوۡفٌۢ بِالۡعِبَادِ یٰۤاَیُّہَا الَّذِیۡنَ اٰمَنُوا ادۡخُلُوۡا فِی السِّلۡمِ کَآفَّۃً ۪ وَ وَّلَا لَا تَتَّبِعُوۡا خُطُوٰتِ الشَّیۡطٰنِ ؕ الشَّیۡطٰنِ ۚ اِنَّہٗ لَکُمۡ عَدُوٌّ مُّبِیۡنٌ فَاِنۡ زَلَلۡتُمۡ مِّنۡۢ بَعۡدِ مَا جَآءَتۡکُمُ الۡبَیِّنٰتُ فَاعۡلَمُوۡۤا اَنَّ اللّٰہَ عَزِیۡزٌ حَکِیۡمٌ ہَلۡ یَنۡظُرُوۡنَ اِلَّاۤ اَنۡ یَّاۡتِیَہُمُ اللّٰہُ فِیۡ ظُلَلٍ مِّنَ الۡغَمَامِ وَ الۡمَلٰٓئِکَۃُ وَ قُضِیَ الۡاَمۡرُ ؕ الۡاَمۡرُ ۚ وَ اِلَی اللّٰہِ تُرۡجَعُ الۡاُمُوۡرُ سَلۡ بَنِیۡۤ اِسۡرَآءِیۡلَ اِسۡرٰٓءِیۡلَ کَمۡ اٰتَیۡنٰہُمۡ مِّنۡ اٰیَۃٍۭ بَیِّنَۃٍ ؕ بَیِّنَۃٍ ۗ وَ مَنۡ یُّبَدِّلۡ نِعۡمَۃَ اللّٰہِ مِنۡۢ بَعۡدِ مَا جَآءَتۡہُ فَاِنَّ اللّٰہَ شَدِیۡدُ الۡعِقَابِ زُیِّنَ لِلَّذِیۡنَ کَفَرُوا الۡحَیٰوۃُ الدُّنۡیَا وَ یَسۡخَرُوۡنَ مِنَ الَّذِیۡنَ اٰمَنُوۡا ۘ اٰمَنُوۡا ۘ وَ الَّذِیۡنَ اتَّقَوۡا فَوۡقَہُمۡ یَوۡمَ الۡقِیٰمَۃِ ؕ الۡقِیٰمَۃِ ۗ وَ اللّٰہُ یَرۡزُقُ مَنۡ یَّشَآءُ بِغَیۡرِ حِسَابٍ کَانَ النَّاسُ اُمَّۃً وَّاحِدَۃً ۟ وّٰحِدَۃً فَبَعَثَ اللّٰہُ النَّبِیّٖنَ مُبَشِّرِیۡنَ وَ مُنۡذِرِیۡنَ ۪ وَ اَنۡزَلَ مَعَہُمُ الۡکِتٰبَ بِالۡحَقِّ لِیَحۡکُمَ بَیۡنَ النَّاسِ فِیۡمَا اخۡتَلَفُوۡا فِیۡہِ ؕ فِیۡہِ ۚ وَ مَا اخۡتَلَفَ فِیۡہِ اِلَّا الَّذِیۡنَ اُوۡتُوۡہُ مِنۡۢ بَعۡدِ مَا جَآءَتۡہُمُ الۡبَیِّنٰتُ بَغۡیًۢا بَیۡنَہُمۡ ۚ بَیۡنَہُمۡ ۖ فَہَدَی اللّٰہُ الَّذِیۡنَ اٰمَنُوۡا لِمَا اخۡتَلَفُوۡا فِیۡہِ مِنَ الۡحَقِّ بِاِذۡنِہٖ ؕ بِاِذۡنِہٖ ۗ وَ اللّٰہُ یَہۡدِیۡ مَنۡ یَّشَآءُ اِلٰی صِرَاطٍ صِرٰطٍ مُّسۡتَقِیۡمٍ اَمۡ حَسِبۡتُمۡ اَنۡ تَدۡخُلُوا الۡجَنَّۃَ وَ لَمَّا یَاۡتِکُمۡ مَّثَلُ الَّذِیۡنَ خَلَوۡا مِنۡ قَبۡلِکُمۡ ؕ مَسَّتۡہُمُ قَبۡلِکُمۡ ۖ مَّسَّتۡہُمُ الۡبَاۡسَآءُ وَ الضَّرَّآءُ وَ زُلۡزِلُوۡا حَتّٰی یَقُوۡلَ الرَّسُوۡلُ وَ الَّذِیۡنَ اٰمَنُوۡا مَعَہٗ مَتٰی نَصۡرُ اللّٰہِ ؕ اللّٰہِ ۗ اَلَاۤ اِنَّ نَصۡرَ اللّٰہِ قَرِیۡبٌ یَسۡـَٔلُوۡنَکَ مَا ذَا مَاذَا یُنۡفِقُوۡنَ ۖ یُنۡفِقُوۡنَ ۬ؕ قُلۡ مَاۤ اَنۡفَقۡتُمۡ مِّنۡ خَیۡرٍ فَلِلۡوَالِدَیۡنِ فَلِلۡوٰلِدَیۡنِ وَ الۡاَقۡرَبِیۡنَ وَ الۡیَتٰمٰی وَ الۡمَسٰکِیۡنِ وَ ابۡنِ‌السَّبِیۡلِ ؕ ابۡنِ السَّبِیۡلِ ۗ وَ مَا تَفۡعَلُوۡا مِنۡ خَیۡرٍ فَاِنَّ اللّٰہَ بِہٖ عَلِیۡمٌ کُتِبَ عَلَیۡکُمُ الۡقِتَالُ وَ ہُوَ کُرۡہٌ لَّکُمۡ ۚ لَّکُمۡ ۖ وَ عَسٰۤی اَنۡ تَکۡرَہُوۡا شَیۡئًا وَّ شَیۡـًٔا وَّہُوَ ہُوَ خَیۡرٌ لَّکُمۡ ۚ لَّکُمۡ ۖ وَ عَسٰۤی اَنۡ تُحِبُّوۡا شَیۡئًا وَّ شَیۡـًٔا وَّہُوَ ہُوَ شَرٌّ لَّکُمۡ ؕ لَّکُمۡ ۗ وَ اللّٰہُ یَعۡلَمُ وَ اَنۡتُمۡ لَا تَعۡلَمُوۡنَ یَسۡـَٔلُوۡنَکَ عَنِ الشَّہۡرِ الۡحَرَامِ قِتَالٍ فِیۡہِ ؕ فِیۡہِ ۖ قُلۡ قِتَالٌ فِیۡہِ کَبِیۡرٌ ؕ وَ کَبِیۡرٌ ۖ وَّصَدٌّ صَدٌّ عَنۡ سَبِیۡلِ اللّٰہِ وَ کُفۡرٌۢ بِہٖ وَ الۡمَسۡجِدِ الۡحَرَامِ ٭ وَ اِخۡرَاجُ اَہۡلِہٖ مِنۡہُ اَکۡبَرُ عِنۡدَ اللّٰہِ ۚ اللّٰہِ ۚ وَ الۡفِتۡنَۃُ اَکۡبَرُ مِنَ الۡقَتۡلِ ؕ الۡقَتۡلِ ۗ وَ لَا یَزَالُوۡنَ یُقَاتِلُوۡنَکُمۡ یُقٰتِلُوۡنَکُمۡ حَتّٰی یَرُدُّوۡکُمۡ عَنۡ دِیۡنِکُمۡ اِنِ اسۡتَطَاعُوۡا ؕ اسۡتَطٰعُوۡا ۚ وَ مَنۡ یَّرۡتَدِدۡ مِنۡکُمۡ عَنۡ دِیۡنِہٖ فَیَمُتۡ وَ ہُوَ کَافِرٌ فَاُولٰٓئِکَ حَبِطَتۡ اَعۡمَالُہُمۡ اَعۡمٰلُہُمۡ فِی الدُّنۡیَا وَ الۡاٰخِرَۃِ ۚ الۡاٰخِرَۃِ ۖ وَ اُولٰٓئِکَ اَصۡحٰبُ النَّارِ ۚ النَّارِ ۖ ہُمۡ فِیۡہَا خٰلِدُوۡنَ اِنَّ الَّذِیۡنَ اٰمَنُوۡا وَ الَّذِیۡنَ ہَاجَرُوۡا وَ جٰہَدُوۡا فِیۡ سَبِیۡلِ اللّٰہِ ۙ اُولٰٓئِکَ یَرۡجُوۡنَ رَحۡمَتَ اللّٰہِ ؕ اللّٰہِ ۚ وَ اللّٰہُ غَفُوۡرٌ رَّحِیۡمٌ ۞ یَسۡـَٔلُوۡنَکَ عَنِ الۡخَمۡرِ وَ الۡمَیۡسِرِؕ الۡمَیۡسِرِ ۖ قُلۡ فِیۡہِمَاۤ اِثۡمٌ کَبِیۡرٌ وَّ وَّمَنٰفِعُ مَنَافِعُ لِلنَّاسِ لِلنَّاسِ ۫ وَ اِثۡمُہُمَاۤ اَکۡبَرُ مِنۡ نَّفۡعِہِمَا ؕ نَّفۡعِہِمَا ۗ وَ یَسۡـَٔلُوۡنَکَ مَا ذَا مَاذَا یُنۡفِقُوۡنَ یُنۡفِقُوۡنَ ۬ؕ قُلِ الۡعَفۡوَ ؕ ۗ کَذٰلِکَ یُبَیِّنُ اللّٰہُ لَکُمُ الۡاٰیٰتِ لَعَلَّکُمۡ تَتَفَکَّرُوۡنَ فِی الدُّنۡیَا وَ الۡاٰخِرَۃِ ؕ الۡاٰخِرَۃِ ۗ وَ یَسۡـَٔلُوۡنَکَ عَنِ الۡیَتٰمٰی ؕ الۡیَتٰمٰی ۖ قُلۡ اِصۡلَاحٌ لَّہُمۡ خَیۡرٌ ؕ وَ خَیۡرٌ ۖ وَّاِنۡ اِنۡ تُخَالِطُوۡہُمۡ فَاِخۡوَانُکُمۡ ؕ فَاِخۡوٰنُکُمۡ ۚ وَ اللّٰہُ یَعۡلَمُ الۡمُفۡسِدَ مِنَ الۡمُصۡلِحِ ؕ الۡمُصۡلِحِ ۚ وَ لَوۡ شَآءَ اللّٰہُ لَاَعۡنَتَکُمۡ ؕ لَاَعۡنَتَکُمۡ ۚ اِنَّ اللّٰہَ عَزِیۡزٌ حَکِیۡمٌ وَ لَا تَنۡکِحُوا الۡمُشۡرِکٰتِ حَتّٰی یُؤۡمِنَّ ؕ یُؤۡمِنَّ ۚ وَ لَاَمَۃٌ مُّؤۡمِنَۃٌ خَیۡرٌ مِّنۡ مُّشۡرِکَۃٍ وَّ وَّلَوۡ لَوۡ اَعۡجَبَتۡکُمۡ ۚ اَعۡجَبَتۡکُمۡ ۗ وَ لَا تُنۡکِحُوا الۡمُشۡرِکِیۡنَ حَتّٰی یُؤۡمِنُوۡا ؕ یُؤۡمِنُوۡا ۚ وَ لَعَبۡدٌ مُّؤۡمِنٌ خَیۡرٌ مِّنۡ مُّشۡرِکٍ وَّ لَوۡ وَّلَوۡ اَعۡجَبَکُمۡ ۗ اَعۡجَبَکُمۡ ؕ اُولٰٓئِکَ یَدۡعُوۡنَ اِلَی النَّارِ ۚۖ النَّارِ ۖ وَ اللّٰہُ یَدۡعُوۡۤا اِلَی الۡجَنَّۃِ وَ الۡمَغۡفِرَۃِ بِاِذۡنِہٖ ۚ بِاِذۡنِہٖ ۖ وَ یُبَیِّنُ اٰیٰتِہٖ لِلنَّاسِ لَعَلَّہُمۡ یَتَذَکَّرُوۡنَ وَ یَسۡـَٔلُوۡنَکَ عَنِ الۡمَحِیۡضِ ؕ الۡمَحِیۡضِ ۖ قُلۡ ہُوَ اَذًی ۙ فَاعۡتَزِلُوا النِّسَآءَ فِی الۡمَحِیۡضِ ۙ ۖ وَ لَا تَقۡرَبُوۡہُنَّ حَتّٰی یَطۡہُرۡنَ ۚ یَطۡہُرۡنَ ۖ فَاِذَا تَطَہَّرۡنَ فَاۡتُوۡہُنَّ مِنۡ حَیۡثُ اَمَرَکُمُ اللّٰہُ ؕ ۚ اِنَّ اللّٰہَ یُحِبُّ التَّوَّابِیۡنَ التَّوّٰبِیۡنَ وَ یُحِبُّ الۡمُتَطَہِّرِیۡنَ نِسَآؤُکُمۡ حَرۡثٌ لَّکُمۡ ۪ فَاۡتُوۡا حَرۡثَکُمۡ اَنّٰی شِئۡتُمۡ ۫ شِئۡتُمۡ ۖ وَ قَدِّمُوۡا لِاَنۡفُسِکُمۡ ؕ لِاَنۡفُسِکُمۡ ۚ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّکُمۡ مُّلٰقُوۡہُ ؕ ۗ وَ بَشِّرِ الۡمُؤۡمِنِیۡنَ وَ لَا تَجۡعَلُوا اللّٰہَ عُرۡضَۃً لِّاَیۡمَانِکُمۡ لِّاَیۡمٰنِکُمۡ اَنۡ تَبَرُّوۡا وَ تَتَّقُوۡا وَ تُصۡلِحُوۡا بَیۡنَ النَّاسِ ؕ النَّاسِ ۗ وَ اللّٰہُ سَمِیۡعٌ عَلِیۡمٌ لَا لَّا یُؤَاخِذُکُمُ اللّٰہُ بِاللَّغۡوِ فِیۡۤ اَیۡمَانِکُمۡ اَیۡمٰنِکُمۡ وَ لٰکِنۡ یُّؤَاخِذُکُمۡ بِمَا کَسَبَتۡ قُلُوۡبُکُمۡ ؕ قُلُوۡبُکُمۡ ۗ وَ اللّٰہُ غَفُوۡرٌ حَلِیۡمٌ لِلَّذِیۡنَ لِّلَّذِیۡنَ یُؤۡلُوۡنَ مِنۡ نِّسَآئِہِمۡ تَرَبُّصُ اَرۡبَعَۃِ اَشۡہُرٍ ۚ اَشۡہُرٍ ۖ فَاِنۡ فَآءُوۡ فَاِنَّ اللّٰہَ غَفُوۡرٌ رَّحِیۡمٌ وَ اِنۡ عَزَمُوا الطَّلَاقَ الطَّلٰقَ فَاِنَّ اللّٰہَ سَمِیۡعٌ عَلِیۡمٌ وَ الۡمُطَلَّقٰتُ یَتَرَبَّصۡنَ بِاَنۡفُسِہِنَّ ثَلٰثَۃَ قُرُوۡٓءٍ ؕ قُرُوۡٓءٍ ۚ وَ لَا یَحِلُّ لَہُنَّ اَنۡ یَّکۡتُمۡنَ مَا خَلَقَ اللّٰہُ فِیۡۤ اَرۡحَامِہِنَّ اِنۡ کُنَّ یُؤۡمِنَّ بِاللّٰہِ وَ الۡیَوۡمِ الۡاٰخِرِ ؕ الۡاٰخِرِ ۚ وَ بُعُوۡلَتُہُنَّ اَحَقُّ بِرَدِّہِنَّ فِیۡ ذٰلِکَ اِنۡ اَرَادُوۡۤا اِصۡلَاحًا ؕ اِصۡلٰحًا ۚ وَ لَہُنَّ مِثۡلُ الَّذِیۡ عَلَیۡہِنَّ بِالۡمَعۡرُوۡفِ ۪ ۚ وَ لِلرِّجَالِ عَلَیۡہِنَّ دَرَجَۃٌ ؕ دَرَجَۃٌ ۗ وَ اللّٰہُ عَزِیۡزٌ حَکِیۡمٌ اَلطَّلَاقُ اَلطَّلٰقُ مَرَّتَانِ ۖ مَرَّتٰنِ۪ فَاِمۡسَاکٌۢ بِمَعۡرُوۡفٍ اَوۡ تَسۡرِیۡحٌۢ بِاِحۡسَانٍ ؕ بِاِحۡسٰنٍ ۗ وَ لَا یَحِلُّ لَکُمۡ اَنۡ تَاۡخُذُوۡا مِمَّاۤ اٰتَیۡتُمُوۡہُنَّ شَیۡئًا شَیۡـًٔا اِلَّاۤ اَنۡ یَّخَافَاۤ اَلَّا یُقِیۡمَا حُدُوۡدَ اللّٰہِ ؕ اللّٰہِ ۖ فَاِنۡ خِفۡتُمۡ اَلَّا یُقِیۡمَا حُدُوۡدَ اللّٰہِ ۙ فَلَا جُنَاحَ عَلَیۡہِمَا فِیۡمَا افۡتَدَتۡ بِہٖ ؕ بِہٖ ۗ تِلۡکَ حُدُوۡدُ اللّٰہِ فَلَا تَعۡتَدُوۡہَا ۚ تَعۡتَدُوۡہَا ۚ وَ مَنۡ یَّتَعَدَّ حُدُوۡدَ اللّٰہِ فَاُولٰٓئِکَ ہُمُ الظّٰلِمُوۡنَ فَاِنۡ طَلَّقَہَا فَلَا تَحِلُّ لَہٗ مِنۡۢ بَعۡدُ حَتّٰی تَنۡکِحَ زَوۡجًا غَیۡرَہٗ ؕ غَیۡرَہٗ ۗ فَاِنۡ طَلَّقَہَا فَلَا جُنَاحَ عَلَیۡہِمَاۤ اَنۡ یَّتَرَاجَعَاۤ اِنۡ ظَنَّاۤ اَنۡ یُّقِیۡمَا حُدُوۡدَ اللّٰہِ ؕ اللّٰہِ ۗ وَ تِلۡکَ حُدُوۡدُ اللّٰہِ یُبَیِّنُہَا لِقَوۡمٍ یَّعۡلَمُوۡنَ وَ اِذَا طَلَّقۡتُمُ النِّسَآءَ فَبَلَغۡنَ اَجَلَہُنَّ فَاَمۡسِکُوۡہُنَّ بِمَعۡرُوۡفٍ اَوۡ سَرِّحُوۡہُنَّ بِمَعۡرُوۡفٍ ۪ ۚ وَ لَا تُمۡسِکُوۡہُنَّ ضِرَارًا لِّتَعۡتَدُوۡا ۚ لِّتَعۡتَدُوۡا ۚ وَ مَنۡ یَّفۡعَلۡ ذٰلِکَ فَقَدۡ ظَلَمَ نَفۡسَہٗ ؕ نَفۡسَہٗ ۚ وَ لَا تَتَّخِذُوۡۤا اٰیٰتِ اللّٰہِ ہُزُوًا ۫ وَّ ہُزُوًا ۚ وَ اذۡکُرُوۡا نِعۡمَتَ اللّٰہِ عَلَیۡکُمۡ وَ مَاۤ اَنۡزَلَ عَلَیۡکُمۡ مِّنَ الۡکِتٰبِ وَ الۡحِکۡمَۃِ یَعِظُکُمۡ بِہٖ ؕ بِہٖ ۚ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ بِکُلِّ شَیۡءٍ عَلِیۡمٌ وَ اِذَا طَلَّقۡتُمُ النِّسَآءَ فَبَلَغۡنَ اَجَلَہُنَّ فَلَا تَعۡضُلُوۡہُنَّ اَنۡ یَّنۡکِحۡنَ اَزۡوَاجَہُنَّ اَزۡوٰجَہُنَّ اِذَا تَرَاضَوۡا تَرٰضَوۡا بَیۡنَہُمۡ بِالۡمَعۡرُوۡفِ ؕ بِالۡمَعۡرُوۡفِ ۗ ذٰلِکَ یُوۡعَظُ بِہٖ مَنۡ کَانَ مِنۡکُمۡ یُؤۡمِنُ بِاللّٰہِ وَ الۡیَوۡمِ الۡاٰخِرِ ؕ الۡاٰخِرِ ۗ ذٰلِکُمۡ اَزۡکٰی لَکُمۡ وَ اَطۡہَرُ ؕ اَطۡہَرُ ۗ وَ اللّٰہُ یَعۡلَمُ وَ اَنۡتُمۡ لَا تَعۡلَمُوۡنَ ۞ وَ الۡوَالِدٰتُ الۡوٰلِدٰتُ یُرۡضِعۡنَ اَوۡلَادَہُنَّ اَوۡلٰدَہُنَّ حَوۡلَیۡنِ کَامِلَیۡنِ ۖ لِمَنۡ اَرَادَ اَنۡ یُّتِمَّ الرَّضَاعَۃَ ؕ الرَّضَاعَۃَ ۚ وَ عَلَی الۡمَوۡلُوۡدِ لَہٗ رِزۡقُہُنَّ وَ کِسۡوَتُہُنَّ کِسۡوَ تُہُنَّ بِالۡمَعۡرُوۡفِ ۚ بِالۡمَعۡرُوۡفِ ؕ لَا تُکَلَّفُ نَفۡسٌ اِلَّا وُسۡعَہَا ۚ وُسۡعَہَا ۚ لَا تُضَآرَّ وَالِدَۃٌۢ بِوَلَدِہَا وٰلِدَۃٌۢ بِوَ لَدِہَا وَ لَا مَوۡلُوۡدٌ لَّہٗ بِوَلَدِہٖ ٭ بِوَ لَدِہٖ ۚ وَ عَلَی الۡوَارِثِ الۡوَ ارِثِ مِثۡلُ ذٰلِکَ ۚ ذٰلِکَ ۗ فَاِنۡ اَرَادَا فِصَالًا عَنۡ تَرَاضٍ مِّنۡہُمَا وَ تَشَاوُرٍ فَلَا جُنَاحَ عَلَیۡہِمَا ؕ عَلَیۡہِمَا ۗ وَ اِنۡ اَرَدۡتُّمۡ اَرَدتُّمۡ اَنۡ تَسۡتَرۡضِعُوۡۤا اَوۡلَادَکُمۡ اَوۡلٰدَکُمۡ فَلَا جُنَاحَ عَلَیۡکُمۡ اِذَا سَلَّمۡتُمۡ مَّاۤ اٰتَیۡتُمۡ بِالۡمَعۡرُوۡفِ ؕ بِالۡمَعۡرُوۡفِ ۗ وَ اتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ بِمَا تَعۡمَلُوۡنَ بَصِیۡرٌ وَ الَّذِیۡنَ یُتَوَفَّوۡنَ یُتَوَ فَّوۡنَ مِنۡکُمۡ وَ یَذَرُوۡنَ اَزۡوَاجًا اَزۡوٰجًا یَّتَرَبَّصۡنَ بِاَنۡفُسِہِنَّ اَرۡبَعَۃَ اَشۡہُرٍ وَّ عَشۡرًا ۚ وَّعَشۡرًا ۖ فَاِذَا بَلَغۡنَ اَجَلَہُنَّ فَلَا جُنَاحَ عَلَیۡکُمۡ فِیۡمَا فِیۡۤ اَنۡفُسِکُمۡ ؕ اَنۡفُسِکُمۡ ۚ عَلِمَ اللّٰہُ اَنَّکُمۡ سَتَذۡکُرُوۡنَہُنَّ وَ لٰکِنۡ لَّا تُوَاعِدُوۡہُنَّ تُوَ اعِدُوۡہُنَّ سِرًّا اِلَّاۤ اَنۡ تَقُوۡلُوۡا قَوۡلًا مَّعۡرُوۡفًا ۬ؕ ۚ وَ لَا تَعۡزِمُوۡا عُقۡدَۃَ النِّکَاحِ حَتّٰی یَبۡلُغَ الۡکِتٰبُ اَجَلَہٗ ؕ اَجَلَہٗ ۚ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ یَعۡلَمُ مَا فِیۡۤ اَنۡفُسِکُمۡ فَاحۡذَرُوۡہُ ۚ فَاحۡذَرُوۡہُ ۚ وَ اعۡلَمُوۡۤا اَنَّ تَمَسُّوۡہُنَّ وَ قَدۡ فَرَضۡتُمۡ لَہُنَّ فَرِیۡضَۃً فَنِصۡفُ مَا فَرَضۡتُمۡ اِلَّاۤ اَنۡ یَّعۡفُوۡنَ اَوۡ یَعۡفُوَا یَعۡفُوَ ا الَّذِیۡ بِیَدِہٖ عُقۡدَۃُ النِّکَاحِ ؕ النِّکَاحِ ۚ وَ اَنۡ تَعۡفُوۡۤا اَقۡرَبُ لِلتَّقۡوٰی ؕ لِلتَّقۡوٰی ۚ وَ لَا تَنۡسَوُا الۡفَضۡلَ بَیۡنَکُمۡ ؕ بَیۡنَکُمۡ ۚ اِنَّ اللّٰہَ بِمَا تَعۡمَلُوۡنَ بَصِیۡرٌ حٰفِظُوۡا عَلَی الصَّلَوٰتِ وَ الصَّلٰوۃِ الۡوُسۡطٰی ٭ وَ قُوۡمُوۡا لِلّٰہِ قٰنِتِیۡنَ فَاِنۡ خِفۡتُمۡ فَرِجَالًا اَوۡ رُکۡبَانًا ۚ رُکۡبَانًا ۖ فَاِذَاۤ اَمِنۡتُمۡ فَاذۡکُرُوا اللّٰہَ کَمَا عَلَّمَکُمۡ مَّا لَمۡ تَکُوۡنُوۡا تَعۡلَمُوۡنَ وَ الَّذِیۡنَ یُتَوَفَّوۡنَ یُتَوَ فَّوۡنَ مِنۡکُمۡ وَ یَذَرُوۡنَ اَزۡوَاجًا ۚۖ اَزۡوٰجًا وَّصِیَّۃً لِّاَزۡوَاجِہِمۡ لِّاَزۡوٰجِہِمۡ مَّتٰعًا مَّتَاعًا اِلَی الۡحَوۡلِ غَیۡرَ اِخۡرَاجٍ ۚ اِخۡرَاجٍ ۚ فَاِنۡ خَرَجۡنَ فَلَا جُنَاحَ عَلَیۡکُمۡ فِیۡ مَا فَعَلۡنَ فِیۡۤ اَنۡفُسِہِنَّ مِنۡ مَّعۡرُوۡفٍ ؕ مَّعۡرُوۡفٍ ۗ وَ اللّٰہُ عَزِیۡزٌ حَکِیۡمٌ وَ لِلۡمُطَلَّقٰتِ مَتَاعٌۢ بِالۡمَعۡرُوۡفِ ؕ مَتٰعٌۢ بِالۡمَعۡرُوۡفِ ۖ حَقًّا فَقَالَ لَہُمُ اللّٰہُ مُوۡتُوۡا ۟ مُوۡتُوۡا ثُمَّ اَحۡیَاہُمۡ اَحۡیٰہُمۡ ۚ ؕ اِنَّ اللّٰہَ لَذُوۡ فَضۡلٍ عَلَی النَّاسِ وَ لٰکِنَّ اَکۡثَرَ النَّاسِ لَا یَشۡکُرُوۡنَ وَ قَاتِلُوۡا قٰتِلُوۡا فِیۡ سَبِیۡلِ اللّٰہِ وَ اعۡلَمُوۡۤا اَنَّ اللّٰہَ سَمِیۡعٌ عَلِیۡمٌ مَنۡ مَّنۡ ذَا الَّذِیۡ یُقۡرِضُ اللّٰہَ قَرۡضًا حَسَنًا فَیُضٰعِفَہٗ لَہٗۤ اَضۡعَافًا کَثِیۡرَۃً ؕ کَثِیۡرَۃً ۚ وَ اللّٰہُ یَقۡبِضُ وَ یَبۡصُۜطُ ۪ وَ اِلَیۡہِ تُرۡجَعُوۡنَ اَلَمۡ تَرَ اِلَی الۡمَلَاِ مِنۡۢ بَنِیۡۤ اِسۡرَآءِیۡلَ اِسۡرٰٓءِیۡلَ مِنۡۢ بَعۡدِ مُوۡسٰی ۘ مُوۡسٰۤی اِذۡ قَالُوۡا لِنَبِیٍّ لَّہُمُ ابۡعَثۡ لَنَا مَلِکًا نُّقَاتِلۡ نُّقٰتِلۡ فِیۡ سَبِیۡلِ اللّٰہِ ؕ اللّٰہِ ۖ قَالَ ہَلۡ عَسَیۡتُمۡ اِنۡ کُتِبَ عَلَیۡکُمُ الۡقِتَالُ اَلَّا تُقَاتِلُوۡا ؕ تُقٰتِلُوۡا ۖ قَالُوۡا وَ مَا لَنَاۤ اَلَّا نُقَاتِلَ نُقٰتِلَ فِیۡ سَبِیۡلِ اللّٰہِ وَ قَدۡ اُخۡرِجۡنَا مِنۡ دِیَارِنَا دِیٰرِنَا وَ اَبۡنَآئِنَا ؕ اَبۡنَآئِنَا ۖ فَلَمَّا کُتِبَ عَلَیۡہِمُ الۡقِتَالُ تَوَلَّوۡا تَوَ لَّوۡا اِلَّا قَلِیۡلًا مِّنۡہُمۡ ؕ مِّنۡہُمۡ ۗ وَ اللّٰہُ عَلِیۡمٌۢ بِالظّٰلِمِیۡنَ وَ قَالَ لَہُمۡ نَبِیُّہُمۡ اِنَّ اللّٰہَ قَدۡ بَعَثَ لَکُمۡ طَالُوۡتَ مَلِکًا ؕ مَلِکًا ۚ قَالُوۡۤا اَنّٰی یَکُوۡنُ لَہُ الۡمُلۡکُ عَلَیۡنَا وَ نَحۡنُ اَحَقُّ مِّمَّا تَرَکَ اٰلُ مُوۡسٰی وَ اٰلُ ہٰرُوۡنَ تَحۡمِلُہُ الۡمَلٰٓئِکَۃُ ؕ الۡمَلٰٓئِکَۃُ ۚ اِنَّ فِیۡ ذٰلِکَ لَاٰیَۃً لَّکُمۡ اِنۡ کُنۡتُمۡ مُّؤۡمِنِیۡنَ فَلَمَّا فَصَلَ طَالُوۡتُ بِالۡجُنُوۡدِ ۙ قَالَ اِنَّ اللّٰہَ مُبۡتَلِیۡکُمۡ بِنَہَرٍ ۚ بِنَہَرٍ فَمَنۡ شَرِبَ مِنۡہُ فَلَیۡسَ مِنِّیۡ ۚ مِنِّیۡ وَ مَنۡ لَّمۡ یَطۡعَمۡہُ فَاِنَّہٗ مِنِّیۡۤ اِلَّا مَنِ اغۡتَرَفَ غُرۡفَۃًۢ بِیَدِہٖ ۚ بِیَدِہٖ ۚ فَشَرِبُوۡا مِنۡہُ اِلَّا قَلِیۡلًا مِّنۡہُمۡ ؕ مِّنۡہُمۡ ۚ فَلَمَّا جَاوَزَہٗ جَاوَ زَہٗ ہُوَ وَ الَّذِیۡنَ اٰمَنُوۡا مَعَہٗ ۙ قَالُوۡا لَا طَاقَۃَ لَنَا الۡیَوۡمَ بِجَالُوۡتَ وَ جُنُوۡدِہٖ ؕ یٰۤاَیُّہَا الَّذِیۡنَ اٰمَنُوۡۤا اِذَا تَدَایَنۡتُمۡ بِدَیۡنٍ اِلٰۤی اَجَلٍ مُّسَمًّی فَاکۡتُبُوۡہُ ؕ فَاکۡتُبُوۡہُ ۚ وَ لۡیَکۡتُبۡ بَّیۡنَکُمۡ کَاتِبٌۢ بِالۡعَدۡلِ ۪ بِالۡعَدۡلِ ۚ وَ لَا یَاۡبَ کَاتِبٌ اَنۡ یَّکۡتُبَ کَمَا عَلَّمَہُ اللّٰہُ فَلۡیَکۡتُبۡ ۚ ۚ فَلۡیَکۡتُبۡ وَ لۡیُمۡلِلِ الَّذِیۡ عَلَیۡہِ الۡحَقُّ وَ لۡیَتَّقِ اللّٰہَ رَبَّہٗ وَ لَا یَبۡخَسۡ مِنۡہُ شَیۡئًا ؕ شَیۡـًٔا ۚ فَاِنۡ کَانَ الَّذِیۡ عَلَیۡہِ الۡحَقُّ سَفِیۡہًا اَوۡ ضَعِیۡفًا اَوۡ لَا یَسۡتَطِیۡعُ اَنۡ یُّمِلَّ ہُوَ فَلۡیُمۡلِلۡ وَلِیُّہٗ بِالۡعَدۡلِ ؕ وَ لِیُّہٗ بِالۡعَدۡلِ ۚ وَ اسۡتَشۡہِدُوۡا شَہِیۡدَیۡنِ مِنۡ رِّجَالِکُمۡ ۚ رِّجَالِکُمۡ ۖ فَاِنۡ لَّمۡ یَکُوۡنَا رَجُلَیۡنِ فَرَجُلٌ وَّ امۡرَاَتٰنِ وَّامۡرَاَتَانِ مِمَّنۡ تَرۡضَوۡنَ مِنَ الشُّہَدَآءِ اَنۡ تَضِلَّ اِحۡدٰٮہُمَا فَتُذَکِّرَ اِحۡدٰٮہُمَا الۡاُخۡرٰی ؕ الۡاُخۡرٰی ۚ وَ لَا یَاۡبَ الشُّہَدَآءُ اِذَا مَا دُعُوۡا ؕ دُعُوۡا ۚ وَ لَا تَسۡـَٔمُوۡۤا اَنۡ تَکۡتُبُوۡہُ صَغِیۡرًا اَوۡ کَبِیۡرًا اِلٰۤی اَجَلِہٖ ؕ اَجَلِہٖ ۚ ذٰلِکُمۡ اَقۡسَطُ عِنۡدَ اللّٰہِ وَ اَقۡوَمُ اَقۡوَ مُ لِلشَّہٰدَۃِ لِلشَّہَادَۃِ وَ اَدۡنٰۤی اَلَّا تَرۡتَابُوۡۤا ۖ اِلَّاۤ اَنۡ تَکُوۡنَ تِجَارَۃً تِجٰرَۃً حَاضِرَۃً تُدِیۡرُوۡنَہَا بَیۡنَکُمۡ فَلَیۡسَ عَلَیۡکُمۡ جُنَاحٌ اَلَّا تَکۡتُبُوۡہَا ؕ تَکۡتُبُوۡہَا ۗ وَ اَشۡہِدُوۡۤا اِذَا تَبَایَعۡتُمۡ ۪ تَبَایَعۡتُمۡ ۚ وَ لَا یُضَآرَّ کَاتِبٌ وَّ لَا وَّلَا شَہِیۡدٌ ۚ شَہِیۡدٌ ۬ؕ وَ اِنۡ تَفۡعَلُوۡا فَاِنَّہٗ فُسُوۡقٌۢ بِکُمۡ ؕ بِکُمۡ ۗ وَ اتَّقُوا اللّٰہَ ؕ اللّٰہَ ۖ وَ یُعَلِّمُکُمُ اللّٰہُ ؕ اللّٰہُ ۗ وَ اللّٰہُ بِکُلِّ شَیۡءٍ عَلِیۡمٌ ۞ وَ اِنۡ کُنۡتُمۡ عَلٰی سَفَرٍ وَّ لَمۡ وَّلَمۡ تَجِدُوۡا کَاتِبًا فَرِہٰنٌ مَّقۡبُوۡضَۃٌ ؕ مَّقۡبُوۡضَۃٌ ۖ فَاِنۡ اَمِنَ بَعۡضُکُمۡ بَعۡضًا فَلۡیُؤَدِّ الَّذِی اؤۡتُمِنَ اَمَانَتَہٗ اَمٰنَتَہٗ وَ لۡیَتَّقِ اللّٰہَ رَبَّہٗ ؕ رَبَّہٗ ۗ وَ لَا تَکۡتُمُوا الشَّہَادَۃَ ؕ الشَّہٰدَۃَ ۚ وَ مَنۡ یَّکۡتُمۡہَا فَاِنَّہٗۤ اٰثِمٌ قَلۡبُہٗ ؕ قَلۡبُہٗ ۗ وَ اللّٰہُ بِمَا تَعۡمَلُوۡنَ عَلِیۡمٌ لِلّٰہِ لِّلّٰہِ مَا فِی السَّمٰوٰتِ وَ مَا فِی الۡاَرۡضِ ؕ الۡاَرۡضِ ۗ وَ اِنۡ تُبۡدُوۡا مَا فِیۡۤ اَنۡفُسِکُمۡ اَوۡ تُخۡفُوۡہُ یُحَاسِبۡکُمۡ بِہِ اللّٰہُ ؕ اللّٰہُ ۖ فَیَغۡفِرُ لِمَنۡ یَّشَآءُ وَ یُعَذِّبُ مَنۡ یَّشَآءُ ؕ یَّشَآءُ ۗ وَ اللّٰہُ عَلٰی کُلِّ شَیۡءٍ قَدِیۡرٌ اٰمَنَ الرَّسُوۡلُ بِمَاۤ اُنۡزِلَ اِلَیۡہِ مِنۡ رَّبِّہٖ وَ الۡمُؤۡمِنُوۡنَ ؕ الۡمُؤۡمِنُوۡنَ ۚ کُلٌّ اٰمَنَ بِاللّٰہِ وَ مَلٰٓئِکَتِہٖ وَ کُتُبِہٖ وَ رُسُلِہٖ ۟ رُسُلِہٖ لَا نُفَرِّقُ بَیۡنَ اَحَدٍ مِّنۡ رُّسُلِہٖ ۟ رُّسُلِہٖ ۚ وَ قَالُوۡا سَمِعۡنَا وَ اَطَعۡنَا ٭۫ ۖ غُفۡرَانَکَ رَبَّنَا وَ اِلَیۡکَ الۡمَصِیۡرُ لَا یُکَلِّفُ اللّٰہُ نَفۡسًا اِلَّا وُسۡعَہَا ؕ وُسۡعَہَا ۚ لَہَا مَا کَسَبَتۡ وَ عَلَیۡہَا مَا اکۡتَسَبَتۡ ؕ اکۡتَسَبَتۡ ۗ رَبَّنَا لَا تُؤَاخِذۡنَاۤ اِنۡ نَّسِیۡنَاۤ اَوۡ اَخۡطَاۡنَا ۚ اَخۡطَاۡنَا ۚ رَبَّنَا وَ لَا تَحۡمِلۡ عَلَیۡنَاۤ اِصۡرًا کَمَا حَمَلۡتَہٗ عَلَی الَّذِیۡنَ مِنۡ قَبۡلِنَا ۚ قَبۡلِنَا ۚ رَبَّنَا وَ لَا تُحَمِّلۡنَا مَا لَا طَاقَۃَ لَنَا بِہٖ ۚ بِہٖ ۖ وَ اعۡفُ عَنَّا ٝ عَنَّا وَ اغۡفِرۡ لَنَا ٝ لَنَا وَ ارۡحَمۡنَا ٝ ارۡحَمۡنَاۤ ۚ اَنۡتَ مَوۡلٰٮنَا فَانۡصُرۡنَا عَلَی الۡقَوۡمِ الۡکٰفِرِیۡنَ بِسۡمِ اللّٰہِ الرَّحۡمٰنِ الرَّحِیۡمِ الٓمَّٓ ۙ اللّٰہُ الٓمّۤ اَللّٰہُ لَاۤ اِلٰہَ اِلَّا ہُوَ ۙ ہُوَ الۡحَیُّ الۡقَیُّوۡمُ ؕ الۡقَیُّوۡمُ نَزَّلَ عَلَیۡکَ الۡکِتٰبَ بِالۡحَقِّ مُصَدِّقًا لِّمَا بَیۡنَ یَدَیۡہِ وَ اَنۡزَلَ التَّوۡرٰٮۃَ وَ الۡاِنۡجِیۡلَ ۙ الۡاِنۡجِیۡلَ مِنۡ قَبۡلُ ہُدًی لِّلنَّاسِ وَ اَنۡزَلَ الۡفُرۡقَانَ ۬ؕ الۡفُرۡقَانَ ۗ اِنَّ الَّذِیۡنَ کَفَرُوۡا بِاٰیٰتِ اللّٰہِ لَہُمۡ عَذَابٌ شَدِیۡدٌ ؕ شَدِیۡدٌ ۗ وَ اللّٰہُ عَزِیۡزٌ ذُو انۡتِقَامٍ اِنَّ اللّٰہَ لَا یَخۡفٰی عَلَیۡہِ شَیۡءٌ فِی الۡاَرۡضِ وَ لَا فِی السَّمَآءِ ؕ
