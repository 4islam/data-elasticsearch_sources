فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ ذَوَاتَاۤ ذَوَ اتَاۤ اَفۡنَانٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِمَا عَیۡنٰنِ عَیۡنَانِ تَجۡرِیٰنِ تَجۡرِیَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِمَا مِنۡ کُلِّ فَاکِہَۃٍ فٰکِہَۃٍ زَوۡجٰنِ زَوۡجَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ مُتَّکِـِٕیۡنَ مُتَّکِئِیۡنَ عَلٰی فُرُشٍۭ بَطَآئِنُہَا مِنۡ اِسۡتَبۡرَقٍ وَ جَنَا جَنَی الۡجَنَّتَیۡنِ دَانٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِنَّ قٰصِرٰتُ الطَّرۡفِ لَمۡ یَطۡمِثۡہُنَّ اِنۡسٌ قَبۡلَہُمۡ وَ لَا جَآنٌّ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ کَاَنَّہُنَّ الۡیَاقُوۡتُ وَ الۡمَرۡجَانُ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ ہَلۡ جَزَآءُ الۡاِحۡسَانِ الۡاِحۡسٰنِ اِلَّا الۡاِحۡسَانُ الۡاِحۡسٰنُ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ وَ مِنۡ دُوۡنِہِمَا جَنَّتٰنِ جَنَّتَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ مُدۡہَآ مَّتٰنِ مُدۡہَآمَّتَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِمَا عَیۡنٰنِ عَیۡنَانِ نَضَّاخَتٰنِ نَضَّاخَتَانِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِمَا فَاکِہَۃٌ فٰکِہَۃٌ وَّ وَّنَخۡلٌ نَخۡلٌ وَّ رُمَّانٌ وَّرُمَّانٌ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ فِیۡہِنَّ خَیۡرٰتٌ حِسَانٌ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ حُوۡرٌ مَّقۡصُوۡرٰتٌ فِی الۡخِیَامِ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ لَمۡ یَطۡمِثۡہُنَّ اِنۡسٌ قَبۡلَہُمۡ وَ لَا جَآنٌّ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ مُتَّکِـِٕیۡنَ مُتَّکِئِیۡنَ عَلٰی رَفۡرَفٍ خُضۡرٍ وَّ عَبۡقَرِیٍّ وَّعَبۡقَرِیٍّ حِسَانٍ فَبِاَیِّ اٰلَآءِ رَبِّکُمَا تُکَذِّبٰنِ تُکَذِّبَانِ تَبٰرَکَ اسۡمُ رَبِّکَ ذِی الۡجَلٰلِ وَ الۡاِکۡرَامِ بِسۡمِ اللّٰہِ الرَّحۡمٰنِ الرَّحِیۡمِ اِذَا وَقَعَتِ وَ قَعَتِ الۡوَاقِعَۃُ الۡوَ اقِعَۃُ لَیۡسَ لِوَقۡعَتِہَا لِوَ قۡعَتِہَا کَاذِبَۃٌ خَافِضَۃٌ رَّافِعَۃٌ اِذَا رُجَّتِ الۡاَرۡضُ رَجًّا وَّ وَ بُسَّتِ الۡجِبَالُ بَسًّا فَکَانَتۡ ہَبَآءً مُّنۡۢبَثًّا وَّ وَ کُنۡتُمۡ اَزۡوَاجًا اَزۡوٰجًا ثَلٰثَۃً فَاَصۡحٰبُ الۡمَیۡمَنَۃِ مَاۤ اَصۡحٰبُ الۡمَیۡمَنَۃِ وَ اَصۡحٰبُ الۡمَشۡـَٔمَۃِ مَاۤ اَصۡحٰبُ الۡمَشۡـَٔمَۃِ وَ السّٰبِقُوۡنَ السّٰبِقُوۡنَ اُولٰٓئِکَ الۡمُقَرَّبُوۡنَ فِیۡ جَنّٰتِ النَّعِیۡمِ ثُلَّۃٌ مِّنَ الۡاَوَّلِیۡنَ وَ قَلِیۡلٌ مِّنَ الۡاٰخِرِیۡنَ عَلٰی سُرُرٍ مَّوۡضُوۡنَۃٍ مُّتَّکِـِٕیۡنَ مُّتَّکِئِیۡنَ عَلَیۡہَا مُتَقٰبِلِیۡنَ یَطُوۡفُ عَلَیۡہِمۡ وِلۡدَانٌ وِلۡدٰنٌ مُّخَلَّدُوۡنَ بِاَکۡوَابٍ بِاَکۡوَ ابٍ وَّ اَبَارِیۡقَ وَّاَبَارِیۡقَ وَ کَاۡسٍ مِّنۡ مَّعِیۡنٍ لَّا یُصَدَّعُوۡنَ عَنۡہَا وَ لَا یُنۡزِفُوۡنَ وَ فَاکِہَۃٍ فٰکِہَۃٍ مِّمَّا یَتَخَیَّرُوۡنَ وَ لَحۡمِ طَیۡرٍ مِّمَّا یَشۡتَہُوۡنَ وَ حُوۡرٌ عِیۡنٌ کَاَمۡثَالِ کَاَمۡثٰلِ اللُّؤۡلُوَٴ اللُّؤۡلُؤِ الۡمَکۡنُوۡنِ جَزَآءًۢ بِمَا کَانُوۡا یَعۡمَلُوۡنَ لَا یَسۡمَعُوۡنَ فِیۡہَا لَغۡوًا وَّ لَا وَّلَا تَاۡثِیۡمًا اِلَّا قِیۡلًا سَلٰمًا سَلٰمًا وَ اَصۡحٰبُ الۡیَمِیۡنِ مَاۤ اَصۡحٰبُ الۡیَمِیۡنِ فِیۡ سِدۡرٍ مَّخۡضُوۡدٍ وَّ وَ طَلۡحٍ مَّنۡضُوۡدٍ وَّ وَ ظِلٍّ مَّمۡدُوۡدٍ وَّ وَ مَآءٍ مَّسۡکُوۡبٍ وَّ وَ فَاکِہَۃٍ فٰکِہَۃٍ کَثِیۡرَۃٍ لَّا مَقۡطُوۡعَۃٍ وَّ لَا وَّلَا مَمۡنُوۡعَۃٍ وَّ وَ فُرُشٍ مَّرۡفُوۡعَۃٍ اِنَّاۤ اَنۡشَاۡنٰہُنَّ اِنۡشَآءً فَجَعَلۡنٰہُنَّ اَبۡکَارًا عُرُبًا اَتۡرَابًا لِّاَصۡحٰبِ الۡیَمِیۡنِ ثُلَّۃٌ مِّنَ الۡاَوَّلِیۡنَ وَ ثُلَّۃٌ مِّنَ الۡاٰخِرِیۡنَ وَ اَصۡحٰبُ الشِّمَالِ مَاۤ اَصۡحٰبُ الشِّمَالِ فِیۡ سَمُوۡمٍ وَّ حَمِیۡمٍ وَّ وَّحَمِیۡمٍ وَ ظِلٍّ مِّنۡ یَّحۡمُوۡمٍ لَّا بَارِدٍ وَّ لَا وَّلَا کَرِیۡمٍ اِنَّہُمۡ کَانُوۡا قَبۡلَ ذٰلِکَ مُتۡرَفِیۡنَ وَ کَانُوۡا یُصِرُّوۡنَ عَلَی الۡحِنۡثِ الۡعَظِیۡمِ وَ کَانُوۡا یَقُوۡلُوۡنَ اَئِذَا مِتۡنَا وَ کُنَّا تُرَابًا وَّ وَّعِظٰمًا عِظَامًا ءَاِنَّا اَءِنَّا لَمَبۡعُوۡثُوۡنَ اَوَ اٰبَآؤُنَا الۡاَوَّلُوۡنَ قُلۡ اِنَّ الۡاَوَّلِیۡنَ وَ الۡاٰخِرِیۡنَ لَمَجۡمُوۡعُوۡنَ اِلٰی مِیۡقَاتِ مِیۡقٰتِ یَوۡمٍ مَّعۡلُوۡمٍ ثُمَّ اِنَّکُمۡ اَیُّہَا الضَّآلُّوۡنَ الۡمُکَذِّبُوۡنَ لَاٰکِلُوۡنَ مِنۡ شَجَرٍ مِّنۡ زَقُّوۡمٍ فَمَالِـُٔوۡنَ مِنۡہَا الۡبُطُوۡنَ فَشٰرِبُوۡنَ عَلَیۡہِ مِنَ الۡحَمِیۡمِ فَشٰرِبُوۡنَ شُرۡبَ الۡہِیۡمِ ہٰذَا نُزُلُہُمۡ یَوۡمَ الدِّیۡنِ نَحۡنُ خَلَقۡنٰکُمۡ فَلَوۡ لَا فَلَوۡلَا تُصَدِّقُوۡنَ اَفَرَءَیۡتُمۡ مَّا تُمۡنُوۡنَ ءَاَنۡتُمۡ تَخۡلُقُوۡنَہٗۤ اَمۡ نَحۡنُ الۡخٰلِقُوۡنَ نَحۡنُ قَدَّرۡنَا بَیۡنَکُمُ الۡمَوۡتَ وَ مَا نَحۡنُ بِمَسۡبُوۡقِیۡنَ عَلٰۤی اَنۡ نُّبَدِّلَ اَمۡثَالَکُمۡ اَمۡثٰلَکُمۡ وَ نُنۡشِئَکُمۡ فِیۡ مَا لَا تَعۡلَمُوۡنَ وَ لَقَدۡ عَلِمۡتُمُ النَّشۡاَۃَ الۡاُوۡلٰی فَلَوۡ لَا فَلَوۡلَا تَذَکَّرُوۡنَ اَفَرَءَیۡتُمۡ مَّا تَحۡرُثُوۡنَ ءَاَنۡتُمۡ تَزۡرَعُوۡنَہٗۤ اَمۡ نَحۡنُ الزّٰرِعُوۡنَ لَوۡ نَشَآءُ لَجَعَلۡنٰہُ حُطَامًا حُطٰمًا فَظَلۡتُمۡ تَفَکَّہُوۡنَ اِنَّا لَمُغۡرَمُوۡنَ بَلۡ نَحۡنُ مَحۡرُوۡمُوۡنَ
