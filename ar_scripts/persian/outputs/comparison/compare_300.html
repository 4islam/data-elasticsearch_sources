وَ قٰتِلُوۡہُمۡ وَقٰتِلُوۡہُمۡ حَتّٰی لَا تَکُوۡنَ فِتۡنَۃٌ وَّ یَکُوۡنَ وَّیَکُوۡنَ الدِّیۡنُ لِلّٰہِ فَاِنِ انۡتَہَوۡا فَلَا عُدۡوَانَ عُدۡوٰنَ اِلَّا عَلَی الظّٰلِمِیۡنَ اَلشَّہۡرُ الشَّہۡرُ الۡحَرَامُ بِالشَّہۡرِ الۡحَرَامِ وَ الۡحُرُمٰتُ وَالۡحُرُمٰتُ قِصَاصٌ فَمَنِ اعۡتَدٰی عَلَیۡکُمۡ فَاعۡتَدُوۡا عَلَیۡہِ بِمِثۡلِ مَا اعۡتَدٰی عَلَیۡکُمۡ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ مَعَ الۡمُتَّقِیۡنَ وَ اَنۡفِقُوۡا وَاَنۡفِقُوۡا فِیۡ سَبِیۡلِ اللّٰہِ وَ لَا وَلَا تُلۡقُوۡا بِاَیۡدِیۡکُمۡ اِلَی التَّہۡلُکَۃِ وَ اَحۡسِنُوۡا وَاَحۡسِنُوۡۤا اِنَّ اللّٰہَ یُحِبُّ الۡمُحۡسِنِیۡنَ وَ اَتِمُّوا وَاَتِمُّوا الۡحَجَّ وَ الۡعُمۡرَۃَ وَالۡعُمۡرَۃَ لِلّٰہِ فَاِنۡ اُحۡصِرۡتُمۡ فَمَا اسۡتَیۡسَرَ مِنَ الۡہَدۡیِ وَ لَا وَلَا تَحۡلِقُوۡا رُءُوۡسَکُمۡ حَتّٰی یَبۡلُغَ الۡہَدۡیُ مَحِلَّہٗ فَمَنۡ کَانَ مِنۡکُمۡ مَّرِیۡضًا اَوۡ بِہٖۤ اَذًی مِّنۡ رَّاۡسِہٖ فَفِدۡیَۃٌ مِّنۡ صِیَامٍ اَوۡ صَدَقَۃٍ اَوۡ نُسُکٍ فَاِذَاۤ اَمِنۡتُمۡ فَمَنۡ تَمَتَّعَ بِالۡعُمۡرَۃِ اِلَی الۡحَجِّ فَمَا اسۡتَیۡسَرَ مِنَ الۡہَدۡیِ فَمَنۡ لَّمۡ یَجِدۡ فَصِیَامُ ثَلٰثَۃِ اَیَّامٍ فِی الۡحَجِّ وَ سَبۡعَۃٍ وَسَبۡعَۃٍ اِذَا رَجَعۡتُمۡ تِلۡکَ عَشَرَۃٌ کَامِلَۃٌ ذٰلِکَ لِمَنۡ لَّمۡ یَکُنۡ اَہۡلُہٗ حَاضِرِی الۡمَسۡجِدِ الۡحَرَامِ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ شَدِیۡدُ الۡعِقَابِ اَلۡحَجُّ الۡحَجُّ اَشۡہُرٌ مَّعۡلُوۡمٰتٌ فَمَنۡ فَرَضَ فِیۡہِنَّ الۡحَجَّ فَلَا رَفَثَ وَ لَا وَلَا فُسُوۡقَ وَ لَا وَلَا جِدَالَ فِی الۡحَجِّ وَ مَا وَمَا تَفۡعَلُوۡا مِنۡ خَیۡرٍ یَّعۡلَمۡہُ اللّٰہُ وَ تَزَوَّدُوۡا وَتَزَوَّدُوۡا فَاِنَّ خَیۡرَ الزَّادِ التَّقۡوٰی وَ اتَّقُوۡنِ وَاتَّقُوۡنِ یٰۤاُولِی الۡاَلۡبَابِ الۡاَلۡبٰبِ لَیۡسَ عَلَیۡکُمۡ جُنَاحٌ اَنۡ تَبۡتَغُوۡا فَضۡلًا مِّنۡ رَّبِّکُمۡ فَاِذَاۤ اَفَضۡتُمۡ مِّنۡ عَرَفٰتٍ فَاذۡکُرُوا اللّٰہَ عِنۡدَ الۡمَشۡعَرِ الۡحَرَامِ وَ اذۡکُرُوۡہُ وَاذۡکُرُوۡہُ کَمَا ہَدٰٮکُمۡ وَ اِنۡ وَاِنۡ کُنۡتُمۡ مِّنۡ قَبۡلِہٖ لَمِنَ الضَّآلِّیۡنَ ثُمَّ اَفِیۡضُوۡا مِنۡ حَیۡثُ اَفَاضَ النَّاسُ وَ اسۡتَغۡفِرُوا وَاسۡتَغۡفِرُوا اللّٰہَ اِنَّ اللّٰہَ غَفُوۡرٌ رَّحِیۡمٌ فَاِذَا قَضَیۡتُمۡ مَّنَاسِکَکُمۡ مَّنٰسِکَکُمۡ فَاذۡکُرُوا اللّٰہَ کَذِکۡرِکُمۡ اٰبَآءَکُمۡ اَوۡ اَشَدَّ ذِکۡرًا فَمِنَ النَّاسِ مَنۡ یَّقُوۡلُ رَبَّنَاۤ اٰتِنَا فِی الدُّنۡیَا وَ مَا وَمَا لَہٗ فِی الۡاٰخِرَۃِ مِنۡ خَلَاقٍ خَلٰقٍ وَ مِنۡہُمۡ وَمِنۡہُمۡ مَّنۡ یَّقُوۡلُ رَبَّنَاۤ اٰتِنَا فِی الدُّنۡیَا حَسَنَۃً وَّ فِی وَّفِی الۡاٰخِرَۃِ حَسَنَۃً وَّ قِنَا وَّقِنَا عَذَابَ النَّارِ اُولٰٓئِکَ لَہُمۡ نَصِیۡبٌ مِّمَّا کَسَبُوۡا وَ اللّٰہُ وَاللّٰہُ سَرِیۡعُ الۡحِسَابِ وَ اذۡکُرُوا وَاذۡکُرُوا اللّٰہَ فِیۡۤ اَیَّامٍ مَّعۡدُوۡدٰتٍ فَمَنۡ تَعَجَّلَ فِیۡ یَوۡمَیۡنِ فَلَاۤ اِثۡمَ عَلَیۡہِ وَ مَنۡ وَمَنۡ تَاَخَّرَ فَلَاۤ اِثۡمَ عَلَیۡہِ لِمَنِ اتَّقٰی وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّکُمۡ اِلَیۡہِ تُحۡشَرُوۡنَ وَ مِنَ وَمِنَ النَّاسِ مَنۡ یُّعۡجِبُکَ قَوۡلُہٗ فِی الۡحَیٰوۃِ الدُّنۡیَا وَ یُشۡہِدُ وَیُشۡہِدُ اللّٰہَ عَلٰی مَا فِیۡ قَلۡبِہٖ وَ ہُوَ وَہُوَ اَلَدُّ الۡخِصَامِ وَ اِذَا وَاِذَا تَوَلّٰی سَعٰی فِی الۡاَرۡضِ لِیُفۡسِدَ فِیۡہَا وَ یُہۡلِکَ وَیُہۡلِکَ الۡحَرۡثَ وَ وَالنَّسۡلَ النَّسۡلَ وَ اللّٰہُ وَاللّٰہُ لَا یُحِبُّ الۡفَسَادَ وَ اِذَا وَاِذَا قِیۡلَ لَہُ اتَّقِ اللّٰہَ اَخَذَتۡہُ الۡعِزَّۃُ بِالۡاِثۡمِ فَحَسۡبُہٗ جَہَنَّمُ وَ لَبِئۡسَ وَلَبِئۡسَ الۡمِہَادُ وَ مِنَ وَمِنَ النَّاسِ مَنۡ یَّشۡرِیۡ نَفۡسَہُ ابۡتِغَآءَ مَرۡضَاتِ اللّٰہِ وَ اللّٰہُ وَاللّٰہُ رَءُوۡفٌۢ بِالۡعِبَادِ یٰۤاَیُّہَا الَّذِیۡنَ اٰمَنُوا ادۡخُلُوۡا فِی السِّلۡمِ کَآفَّۃً وَ لَا وَّلَا تَتَّبِعُوۡا خُطُوٰتِ الشَّیۡطٰنِ اِنَّہٗ لَکُمۡ عَدُوٌّ مُّبِیۡنٌ فَاِنۡ زَلَلۡتُمۡ مِّنۡۢ بَعۡدِ مَا جَآءَتۡکُمُ الۡبَیِّنٰتُ فَاعۡلَمُوۡۤا اَنَّ اللّٰہَ عَزِیۡزٌ حَکِیۡمٌ ہَلۡ یَنۡظُرُوۡنَ اِلَّاۤ اَنۡ یَّاۡتِیَہُمُ اللّٰہُ فِیۡ ظُلَلٍ مِّنَ الۡغَمَامِ وَ وَالۡمَلٰٓئِکَۃُ الۡمَلٰٓئِکَۃُ وَ قُضِیَ وَقُضِیَ الۡاَمۡرُ وَ اِلَی وَاِلَی اللّٰہِ تُرۡجَعُ الۡاُمُوۡرُ سَلۡ بَنِیۡۤ اِسۡرَآءِیۡلَ اِسۡرٰٓءِیۡلَ کَمۡ اٰتَیۡنٰہُمۡ مِّنۡ اٰیَۃٍۭ بَیِّنَۃٍ وَ مَنۡ وَمَنۡ یُّبَدِّلۡ نِعۡمَۃَ اللّٰہِ مِنۡۢ بَعۡدِ مَا جَآءَتۡہُ فَاِنَّ اللّٰہَ شَدِیۡدُ الۡعِقَابِ زُیِّنَ لِلَّذِیۡنَ کَفَرُوا الۡحَیٰوۃُ الدُّنۡیَا وَ یَسۡخَرُوۡنَ وَیَسۡخَرُوۡنَ مِنَ الَّذِیۡنَ اٰمَنُوۡا وَ الَّذِیۡنَ وَالَّذِیۡنَ اتَّقَوۡا فَوۡقَہُمۡ یَوۡمَ الۡقِیٰمَۃِ وَ اللّٰہُ وَاللّٰہُ یَرۡزُقُ مَنۡ یَّشَآءُ بِغَیۡرِ حِسَابٍ کَانَ النَّاسُ اُمَّۃً وَّاحِدَۃً وّٰحِدَۃً فَبَعَثَ اللّٰہُ النَّبِیّٖنَ مُبَشِّرِیۡنَ وَ وَمُنۡذِرِیۡنَ مُنۡذِرِیۡنَ وَ اَنۡزَلَ وَاَنۡزَلَ مَعَہُمُ الۡکِتٰبَ بِالۡحَقِّ لِیَحۡکُمَ بَیۡنَ النَّاسِ فِیۡمَا اخۡتَلَفُوۡا فِیۡہِ وَ مَا وَمَا اخۡتَلَفَ فِیۡہِ اِلَّا الَّذِیۡنَ اُوۡتُوۡہُ مِنۡۢ بَعۡدِ مَا جَآءَتۡہُمُ الۡبَیِّنٰتُ بَغۡیًۢا بَیۡنَہُمۡ فَہَدَی اللّٰہُ الَّذِیۡنَ اٰمَنُوۡا لِمَا اخۡتَلَفُوۡا فِیۡہِ مِنَ الۡحَقِّ بِاِذۡنِہٖ وَ اللّٰہُ وَاللّٰہُ یَہۡدِیۡ مَنۡ یَّشَآءُ اِلٰی صِرَاطٍ صِرٰطٍ مُّسۡتَقِیۡمٍ اَمۡ حَسِبۡتُمۡ اَنۡ تَدۡخُلُوا الۡجَنَّۃَ وَ لَمَّا وَلَمَّا یَاۡتِکُمۡ مَّثَلُ الَّذِیۡنَ خَلَوۡا مِنۡ قَبۡلِکُمۡ مَسَّتۡہُمُ مَّسَّتۡہُمُ الۡبَاۡسَآءُ وَ وَالضَّرَّآءُ الضَّرَّآءُ وَ زُلۡزِلُوۡا وَزُلۡزِلُوۡا حَتّٰی یَقُوۡلَ الرَّسُوۡلُ وَ الَّذِیۡنَ وَالَّذِیۡنَ اٰمَنُوۡا مَعَہٗ مَتٰی نَصۡرُ اللّٰہِ اَلَاۤ اِنَّ نَصۡرَ اللّٰہِ قَرِیۡبٌ یَسۡـَٔلُوۡنَکَ مَا ذَا مَاذَا یُنۡفِقُوۡنَ قُلۡ مَاۤ اَنۡفَقۡتُمۡ مِّنۡ خَیۡرٍ فَلِلۡوَالِدَیۡنِ فَلِلۡوٰلِدَیۡنِ وَ وَالۡاَقۡرَبِیۡنَ الۡاَقۡرَبِیۡنَ وَالۡیَتٰمٰی وَ وَالۡمَسٰکِیۡنِ الۡیَتٰمٰی وَابۡنِ وَ السَّبِیۡلِ الۡمَسٰکِیۡنِ وَ ابۡنِ‌السَّبِیۡلِ وَ مَا وَمَا تَفۡعَلُوۡا مِنۡ خَیۡرٍ فَاِنَّ اللّٰہَ بِہٖ عَلِیۡمٌ کُتِبَ عَلَیۡکُمُ الۡقِتَالُ وَ ہُوَ وَہُوَ کُرۡہٌ لَّکُمۡ وَ عَسٰۤی وَعَسٰۤی اَنۡ تَکۡرَہُوۡا شَیۡئًا شَیۡـًٔا وَّ ہُوَ وَّہُوَ خَیۡرٌ لَّکُمۡ وَ عَسٰۤی وَعَسٰۤی اَنۡ تُحِبُّوۡا شَیۡئًا شَیۡـًٔا وَّ ہُوَ وَّہُوَ شَرٌّ لَّکُمۡ وَ اللّٰہُ وَاللّٰہُ یَعۡلَمُ وَ اَنۡتُمۡ وَاَنۡتُمۡ لَا تَعۡلَمُوۡنَ یَسۡـَٔلُوۡنَکَ عَنِ الشَّہۡرِ الۡحَرَامِ قِتَالٍ فِیۡہِ قُلۡ قِتَالٌ فِیۡہِ کَبِیۡرٌ وَ صَدٌّ وَّصَدٌّ عَنۡ سَبِیۡلِ اللّٰہِ وَ کُفۡرٌۢ وَکُفۡرٌۢ بِہٖ وَ الۡمَسۡجِدِ وَالۡمَسۡجِدِ الۡحَرَامِ وَ اِخۡرَاجُ وَاِخۡرَاجُ اَہۡلِہٖ مِنۡہُ اَکۡبَرُ عِنۡدَ اللّٰہِ وَ الۡفِتۡنَۃُ وَالۡفِتۡنَۃُ اَکۡبَرُ مِنَ الۡقَتۡلِ وَ لَا وَلَا یَزَالُوۡنَ یُقَاتِلُوۡنَکُمۡ یُقٰتِلُوۡنَکُمۡ حَتّٰی یَرُدُّوۡکُمۡ عَنۡ دِیۡنِکُمۡ اِنِ اسۡتَطَاعُوۡا اسۡتَطٰعُوۡا وَ مَنۡ وَمَنۡ یَّرۡتَدِدۡ مِنۡکُمۡ عَنۡ دِیۡنِہٖ فَیَمُتۡ وَ ہُوَ وَہُوَ کَافِرٌ فَاُولٰٓئِکَ حَبِطَتۡ اَعۡمَالُہُمۡ اَعۡمٰلُہُمۡ فِی الدُّنۡیَا وَ وَالۡاٰخِرَۃِ الۡاٰخِرَۃِ وَ اُولٰٓئِکَ وَاُولٰٓئِکَ اَصۡحٰبُ النَّارِ ہُمۡ فِیۡہَا خٰلِدُوۡنَ اِنَّ الَّذِیۡنَ اٰمَنُوۡا وَ الَّذِیۡنَ وَالَّذِیۡنَ ہَاجَرُوۡا وَ جٰہَدُوۡا وَجٰہَدُوۡا فِیۡ سَبِیۡلِ اللّٰہِ اُولٰٓئِکَ یَرۡجُوۡنَ رَحۡمَتَ اللّٰہِ وَ اللّٰہُ وَاللّٰہُ غَفُوۡرٌ رَّحِیۡمٌ یَسۡـَٔلُوۡنَکَ عَنِ الۡخَمۡرِ وَ الۡمَیۡسِرِؕ وَالۡمَیۡسِرِ قُلۡ فِیۡہِمَاۤ اِثۡمٌ کَبِیۡرٌ وَّ مَنَافِعُ وَّمَنٰفِعُ لِلنَّاسِ وَ اِثۡمُہُمَاۤ وَاِثۡمُہُمَاۤ اَکۡبَرُ مِنۡ نَّفۡعِہِمَا وَ وَیَسۡـَٔلُوۡنَکَ یَسۡـَٔلُوۡنَکَ مَا ذَا مَاذَا یُنۡفِقُوۡنَ قُلِ الۡعَفۡوَ کَذٰلِکَ یُبَیِّنُ اللّٰہُ لَکُمُ الۡاٰیٰتِ لَعَلَّکُمۡ تَتَفَکَّرُوۡنَ فِی الدُّنۡیَا وَ وَالۡاٰخِرَۃِ الۡاٰخِرَۃِ وَ یَسۡـَٔلُوۡنَکَ وَیَسۡـَٔلُوۡنَکَ عَنِ الۡیَتٰمٰی قُلۡ اِصۡلَاحٌ لَّہُمۡ خَیۡرٌ وَ اِنۡ وَّاِنۡ تُخَالِطُوۡہُمۡ فَاِخۡوَانُکُمۡ فَاِخۡوٰنُکُمۡ وَ اللّٰہُ وَاللّٰہُ یَعۡلَمُ الۡمُفۡسِدَ مِنَ الۡمُصۡلِحِ وَ لَوۡ وَلَوۡ شَآءَ اللّٰہُ لَاَعۡنَتَکُمۡ اِنَّ اللّٰہَ عَزِیۡزٌ حَکِیۡمٌ وَ لَا وَلَا تَنۡکِحُوا الۡمُشۡرِکٰتِ حَتّٰی یُؤۡمِنَّ وَ لَاَمَۃٌ وَلَاَمَۃٌ مُّؤۡمِنَۃٌ خَیۡرٌ مِّنۡ مُّشۡرِکَۃٍ وَّ لَوۡ وَّلَوۡ اَعۡجَبَتۡکُمۡ وَ لَا وَلَا تُنۡکِحُوا الۡمُشۡرِکِیۡنَ حَتّٰی یُؤۡمِنُوۡا وَ لَعَبۡدٌ وَلَعَبۡدٌ مُّؤۡمِنٌ خَیۡرٌ مِّنۡ مُّشۡرِکٍ وَّ لَوۡ وَّلَوۡ اَعۡجَبَکُمۡ اُولٰٓئِکَ یَدۡعُوۡنَ اِلَی النَّارِ وَ اللّٰہُ وَاللّٰہُ یَدۡعُوۡۤا اِلَی الۡجَنَّۃِ وَ الۡمَغۡفِرَۃِ وَالۡمَغۡفِرَۃِ بِاِذۡنِہٖ وَ یُبَیِّنُ وَیُبَیِّنُ اٰیٰتِہٖ لِلنَّاسِ لَعَلَّہُمۡ یَتَذَکَّرُوۡنَ وَ یَسۡـَٔلُوۡنَکَ وَیَسۡـَٔلُوۡنَکَ عَنِ الۡمَحِیۡضِ قُلۡ ہُوَ اَذًی فَاعۡتَزِلُوا النِّسَآءَ فِی الۡمَحِیۡضِ وَ لَا وَلَا تَقۡرَبُوۡہُنَّ حَتّٰی یَطۡہُرۡنَ فَاِذَا تَطَہَّرۡنَ فَاۡتُوۡہُنَّ مِنۡ حَیۡثُ اَمَرَکُمُ اللّٰہُ اِنَّ اللّٰہَ یُحِبُّ التَّوَّابِیۡنَ التَّوّٰبِیۡنَ وَ یُحِبُّ وَیُحِبُّ الۡمُتَطَہِّرِیۡنَ نِسَآؤُکُمۡ حَرۡثٌ لَّکُمۡ فَاۡتُوۡا حَرۡثَکُمۡ اَنّٰی شِئۡتُمۡ وَ قَدِّمُوۡا وَقَدِّمُوۡا لِاَنۡفُسِکُمۡ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّکُمۡ مُّلٰقُوۡہُ وَ بَشِّرِ وَبَشِّرِ الۡمُؤۡمِنِیۡنَ وَ لَا وَلَا تَجۡعَلُوا اللّٰہَ عُرۡضَۃً لِّاَیۡمَانِکُمۡ لِّاَیۡمٰنِکُمۡ اَنۡ تَبَرُّوۡا وَ وَتَتَّقُوۡا تَتَّقُوۡا وَ تُصۡلِحُوۡا وَتُصۡلِحُوۡا بَیۡنَ النَّاسِ وَ اللّٰہُ وَاللّٰہُ سَمِیۡعٌ عَلِیۡمٌ لَا لَّا یُؤَاخِذُکُمُ اللّٰہُ بِاللَّغۡوِ فِیۡۤ اَیۡمَانِکُمۡ اَیۡمٰنِکُمۡ وَ لٰکِنۡ وَلٰکِنۡ یُّؤَاخِذُکُمۡ بِمَا کَسَبَتۡ قُلُوۡبُکُمۡ وَ اللّٰہُ وَاللّٰہُ غَفُوۡرٌ حَلِیۡمٌ لِلَّذِیۡنَ لِّلَّذِیۡنَ یُؤۡلُوۡنَ مِنۡ نِّسَآئِہِمۡ تَرَبُّصُ اَرۡبَعَۃِ اَشۡہُرٍ فَاِنۡ فَآءُوۡ فَاِنَّ اللّٰہَ غَفُوۡرٌ رَّحِیۡمٌ وَ اِنۡ وَاِنۡ عَزَمُوا الطَّلَاقَ الطَّلٰقَ فَاِنَّ اللّٰہَ سَمِیۡعٌ عَلِیۡمٌ وَ الۡمُطَلَّقٰتُ وَالۡمُطَلَّقٰتُ یَتَرَبَّصۡنَ بِاَنۡفُسِہِنَّ ثَلٰثَۃَ قُرُوۡٓءٍ وَ لَا وَلَا یَحِلُّ لَہُنَّ اَنۡ یَّکۡتُمۡنَ مَا خَلَقَ اللّٰہُ فِیۡۤ اَرۡحَامِہِنَّ اِنۡ کُنَّ یُؤۡمِنَّ بِاللّٰہِ وَ الۡیَوۡمِ وَالۡیَوۡمِ الۡاٰخِرِ وَ بُعُوۡلَتُہُنَّ وَبُعُوۡلَتُہُنَّ اَحَقُّ بِرَدِّہِنَّ فِیۡ ذٰلِکَ اِنۡ اَرَادُوۡۤا اِصۡلَاحًا اِصۡلٰحًا وَ لَہُنَّ وَلَہُنَّ مِثۡلُ الَّذِیۡ عَلَیۡہِنَّ بِالۡمَعۡرُوۡفِ وَ لِلرِّجَالِ وَلِلرِّجَالِ عَلَیۡہِنَّ دَرَجَۃٌ وَ اللّٰہُ وَاللّٰہُ عَزِیۡزٌ حَکِیۡمٌ اَلطَّلَاقُ الطَّلٰقُ مَرَّتٰنِ۪ مَرَّتَانِ فَاِمۡسَاکٌۢ بِمَعۡرُوۡفٍ اَوۡ تَسۡرِیۡحٌۢ بِاِحۡسَانٍ بِاِحۡسٰنٍ وَ لَا وَلَا یَحِلُّ لَکُمۡ اَنۡ تَاۡخُذُوۡا مِمَّاۤ اٰتَیۡتُمُوۡہُنَّ شَیۡئًا شَیۡـًٔا اِلَّاۤ اَنۡ یَّخَافَاۤ اَلَّا یُقِیۡمَا حُدُوۡدَ اللّٰہِ فَاِنۡ خِفۡتُمۡ اَلَّا یُقِیۡمَا حُدُوۡدَ اللّٰہِ فَلَا جُنَاحَ عَلَیۡہِمَا فِیۡمَا افۡتَدَتۡ بِہٖ تِلۡکَ حُدُوۡدُ اللّٰہِ فَلَا تَعۡتَدُوۡہَا وَ مَنۡ وَمَنۡ یَّتَعَدَّ حُدُوۡدَ اللّٰہِ فَاُولٰٓئِکَ ہُمُ الظّٰلِمُوۡنَ فَاِنۡ طَلَّقَہَا فَلَا تَحِلُّ لَہٗ مِنۡۢ بَعۡدُ حَتّٰی تَنۡکِحَ زَوۡجًا غَیۡرَہٗ فَاِنۡ طَلَّقَہَا فَلَا جُنَاحَ عَلَیۡہِمَاۤ اَنۡ یَّتَرَاجَعَاۤ اِنۡ ظَنَّاۤ اَنۡ یُّقِیۡمَا حُدُوۡدَ اللّٰہِ وَ تِلۡکَ وَتِلۡکَ حُدُوۡدُ اللّٰہِ یُبَیِّنُہَا لِقَوۡمٍ یَّعۡلَمُوۡنَ وَ اِذَا وَاِذَا طَلَّقۡتُمُ النِّسَآءَ فَبَلَغۡنَ اَجَلَہُنَّ فَاَمۡسِکُوۡہُنَّ بِمَعۡرُوۡفٍ اَوۡ سَرِّحُوۡہُنَّ بِمَعۡرُوۡفٍ وَ لَا وَلَا تُمۡسِکُوۡہُنَّ ضِرَارًا لِّتَعۡتَدُوۡا وَ مَنۡ وَمَنۡ یَّفۡعَلۡ ذٰلِکَ فَقَدۡ ظَلَمَ نَفۡسَہٗ وَ لَا وَلَا تَتَّخِذُوۡۤا اٰیٰتِ اللّٰہِ ہُزُوًا وَّ اذۡکُرُوۡا وَاذۡکُرُوۡا نِعۡمَتَ اللّٰہِ عَلَیۡکُمۡ وَ مَاۤ وَمَاۤ اَنۡزَلَ عَلَیۡکُمۡ مِّنَ الۡکِتٰبِ وَ الۡحِکۡمَۃِ وَالۡحِکۡمَۃِ یَعِظُکُمۡ بِہٖ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ بِکُلِّ شَیۡءٍ عَلِیۡمٌ وَ اِذَا وَاِذَا طَلَّقۡتُمُ النِّسَآءَ فَبَلَغۡنَ اَجَلَہُنَّ فَلَا تَعۡضُلُوۡہُنَّ اَنۡ یَّنۡکِحۡنَ اَزۡوَاجَہُنَّ اَزۡوٰجَہُنَّ اِذَا تَرَاضَوۡا تَرٰضَوۡا بَیۡنَہُمۡ بِالۡمَعۡرُوۡفِ ذٰلِکَ یُوۡعَظُ بِہٖ مَنۡ کَانَ مِنۡکُمۡ یُؤۡمِنُ بِاللّٰہِ وَ الۡیَوۡمِ وَالۡیَوۡمِ الۡاٰخِرِ ذٰلِکُمۡ اَزۡکٰی لَکُمۡ وَ وَاَطۡہَرُ اَطۡہَرُ وَ اللّٰہُ وَاللّٰہُ یَعۡلَمُ وَ اَنۡتُمۡ وَاَنۡتُمۡ لَا تَعۡلَمُوۡنَ وَ الۡوَالِدٰتُ وَالۡوٰلِدٰتُ یُرۡضِعۡنَ اَوۡلَادَہُنَّ اَوۡلٰدَہُنَّ حَوۡلَیۡنِ کَامِلَیۡنِ لِمَنۡ اَرَادَ اَنۡ یُّتِمَّ الرَّضَاعَۃَ وَ عَلَی وَعَلَی الۡمَوۡلُوۡدِ لَہٗ رِزۡقُہُنَّ وَ کِسۡوَتُہُنَّ وَکِسۡوَتُہُنَّ بِالۡمَعۡرُوۡفِ لَا تُکَلَّفُ نَفۡسٌ اِلَّا وُسۡعَہَا لَا تُضَآرَّ وَالِدَۃٌۢ وٰلِدَۃٌۢ بِوَلَدِہَا وَ لَا وَلَا مَوۡلُوۡدٌ لَّہٗ بِوَلَدِہٖ وَ عَلَی وَعَلَی الۡوَارِثِ مِثۡلُ ذٰلِکَ فَاِنۡ اَرَادَا فِصَالًا عَنۡ تَرَاضٍ مِّنۡہُمَا وَ تَشَاوُرٍ وَتَشَاوُرٍ فَلَا جُنَاحَ عَلَیۡہِمَا وَ وَاِنۡ اِنۡ اَرَدۡتُّمۡ اَرَدتُّمۡ اَنۡ تَسۡتَرۡضِعُوۡۤا اَوۡلَادَکُمۡ اَوۡلٰدَکُمۡ فَلَا جُنَاحَ عَلَیۡکُمۡ اِذَا سَلَّمۡتُمۡ مَّاۤ اٰتَیۡتُمۡ بِالۡمَعۡرُوۡفِ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ بِمَا تَعۡمَلُوۡنَ بَصِیۡرٌ وَ الَّذِیۡنَ وَالَّذِیۡنَ یُتَوَفَّوۡنَ مِنۡکُمۡ وَ وَیَذَرُوۡنَ یَذَرُوۡنَ اَزۡوَاجًا اَزۡوٰجًا یَّتَرَبَّصۡنَ بِاَنۡفُسِہِنَّ اَرۡبَعَۃَ اَشۡہُرٍ وَّ عَشۡرًا وَّعَشۡرًا فَاِذَا بَلَغۡنَ اَجَلَہُنَّ فَلَا جُنَاحَ عَلَیۡکُمۡ فِیۡمَا فَعَلۡنَ فِیۡۤ اَنۡفُسِہِنَّ بِالۡمَعۡرُوۡفِ وَ اللّٰہُ وَاللّٰہُ بِمَا تَعۡمَلُوۡنَ خَبِیۡرٌ وَ لَا وَلَا جُنَاحَ عَلَیۡکُمۡ فِیۡمَا عَرَّضۡتُمۡ بِہٖ مِنۡ خِطۡبَۃِ النِّسَآءِ اَوۡ اَکۡنَنۡتُمۡ فِیۡۤ اَنۡفُسِکُمۡ عَلِمَ اللّٰہُ اَنَّکُمۡ سَتَذۡکُرُوۡنَہُنَّ وَ لٰکِنۡ وَلٰکِنۡ لَّا تُوَاعِدُوۡہُنَّ سِرًّا اِلَّاۤ اَنۡ تَقُوۡلُوۡا قَوۡلًا مَّعۡرُوۡفًا وَ لَا وَلَا تَعۡزِمُوۡا عُقۡدَۃَ النِّکَاحِ حَتّٰی یَبۡلُغَ الۡکِتٰبُ اَجَلَہٗ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ یَعۡلَمُ مَا فِیۡۤ اَنۡفُسِکُمۡ فَاحۡذَرُوۡہُ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ غَفُوۡرٌ حَلِیۡمٌ لَا لَّا جُنَاحَ عَلَیۡکُمۡ اِنۡ طَلَّقۡتُمُ النِّسَآءَ مَا لَمۡ تَمَسُّوۡہُنَّ اَوۡ تَفۡرِضُوۡا لَہُنَّ فَرِیۡضَۃً وَّ مَتِّعُوۡہُنَّ وَمَتِّعُوۡہُنَّ عَلَی الۡمُوۡسِعِ قَدَرُہٗ وَ عَلَی وَعَلَی الۡمُقۡتِرِ قَدَرُہٗ مَتَاعًۢا مَتٰعًۢا بِالۡمَعۡرُوۡفِ حَقًّا عَلَی الۡمُحۡسِنِیۡنَ وَ اِنۡ وَاِنۡ طَلَّقۡتُمُوۡہُنَّ مِنۡ قَبۡلِ اَنۡ تَمَسُّوۡہُنَّ وَ قَدۡ وَقَدۡ فَرَضۡتُمۡ لَہُنَّ فَرِیۡضَۃً فَنِصۡفُ مَا فَرَضۡتُمۡ اِلَّاۤ اَنۡ یَّعۡفُوۡنَ اَوۡ یَعۡفُوَا الَّذِیۡ بِیَدِہٖ عُقۡدَۃُ طَعَامِکَ وَ شَرَابِکَ وَشَرَابِکَ لَمۡ یَتَسَنَّہۡ وَ انۡظُرۡ وَانۡظُرۡ اِلٰی حِمَارِکَ وَ لِنَجۡعَلَکَ وَلِنَجۡعَلَکَ اٰیَۃً لِّلنَّاسِ وَ انۡظُرۡ وَانۡظُرۡ اِلَی الۡعِظَامِ کَیۡفَ نُنۡشِزُہَا ثُمَّ نَکۡسُوۡہَا لَحۡمًا فَلَمَّا تَبَیَّنَ لَہٗ قَالَ اَعۡلَمُ اَنَّ اللّٰہَ عَلٰی کُلِّ شَیۡءٍ قَدِیۡرٌ وَ اِذۡ وَاِذۡ قَالَ اِبۡرٰہٖمُ رَبِّ اَرِنِیۡ کَیۡفَ تُحۡیِ الۡمَوۡتٰی قَالَ اَوَ لَمۡ اَوَلَمۡ سَنَابِلَ فِیۡ کُلِّ سُنۡۢبُلَۃٍ مِّائَۃُ حَبَّۃٍ وَ اللّٰہُ وَاللّٰہُ یُضٰعِفُ لِمَنۡ یَّشَآءُ وَ وَاللّٰہُ اللّٰہُ وَاسِعٌ وٰسِعٌ عَلِیۡمٌ اَلَّذِیۡنَ الَّذِیۡنَ یُنۡفِقُوۡنَ اَمۡوَالَہُمۡ اَمۡوٰلَہُمۡ فِیۡ سَبِیۡلِ اللّٰہِ ثُمَّ لَا یُتۡبِعُوۡنَ مَاۤ اَنۡفَقُوۡا مَنًّا وَّ لَاۤ وَّلَاۤ اَذًی لَّہُمۡ اَجۡرُہُمۡ عِنۡدَ رَبِّہِمۡ وَ لَا وَلَا خَوۡفٌ عَلَیۡہِمۡ وَ لَا وَلَا ہُمۡ یَحۡزَنُوۡنَ قَوۡلٌ مَّعۡرُوۡفٌ وَّ مَغۡفِرَۃٌ وَّمَغۡفِرَۃٌ خَیۡرٌ مِّنۡ صَدَقَۃٍ یَّتۡبَعُہَاۤ اَذًی وَ اللّٰہُ وَاللّٰہُ غَنِیٌّ حَلِیۡمٌ یٰۤاَیُّہَا الَّذِیۡنَ اٰمَنُوۡا لَا تُبۡطِلُوۡا صَدَقٰتِکُمۡ بِالۡمَنِّ وَ الۡاَذٰی وَالۡاَذٰی کَالَّذِیۡ یُنۡفِقُ مَالَہٗ رِئَآءَ النَّاسِ وَ لَا وَلَا یُؤۡمِنُ بِاللّٰہِ وَ وَالۡیَوۡمِ الۡیَوۡمِ الۡاٰخِرِؕ الۡاٰخِرِ فَمَثَلُہٗ کَمَثَلِ صَفۡوَانٍ عَلَیۡہِ تُرَابٌ فَاَصَابَہٗ وَابِلٌ فَتَرَکَہٗ صَلۡدًا لَا لَّا یَقۡدِرُوۡنَ عَلٰی شَیۡءٍ مِّمَّا کَسَبُوۡا وَ اللّٰہُ وَاللّٰہُ لَا یَہۡدِی الۡقَوۡمَ الۡکٰفِرِیۡنَ وَ مَثَلُ وَمَثَلُ الَّذِیۡنَ یُنۡفِقُوۡنَ اَمۡوَالَہُمُ اَمۡوٰلَہُمُ ابۡتِغَآءَ مَرۡضَاتِ اللّٰہِ وَ تَثۡبِیۡتًا وَتَثۡبِیۡتًا مِّنۡ اَنۡفُسِہِمۡ کَمَثَلِ جَنَّۃٍۭ بِرَبۡوَۃٍ اَصَابَہَا وَابِلٌ فَاٰتَتۡ اُکُلَہَا ضِعۡفَیۡنِ فَاِنۡ لَّمۡ یُصِبۡہَا وَابِلٌ فَطَلٌّ وَ اللّٰہُ وَاللّٰہُ بِمَا تَعۡمَلُوۡنَ بَصِیۡرٌ اَیَوَدُّ اَحَدُکُمۡ اَنۡ تَکُوۡنَ لَہٗ جَنَّۃٌ مِّنۡ نَّخِیۡلٍ وَّ اَعۡنَابٍ وَّاَعۡنَابٍ تَجۡرِیۡ مِنۡ اللّٰہُ لَکُمُ الۡاٰیٰتِ لَعَلَّکُمۡ تَتَفَکَّرُوۡنَ یٰۤاَیُّہَا الَّذِیۡنَ اٰمَنُوۡۤا اَنۡفِقُوۡا مِنۡ طَیِّبٰتِ مَا کَسَبۡتُمۡ وَ مِمَّاۤ وَمِمَّاۤ اَخۡرَجۡنَا لَکُمۡ مِّنَ الۡاَرۡضِ وَ لَا وَلَا تَیَمَّمُوا الۡخَبِیۡثَ مِنۡہُ تُنۡفِقُوۡنَ وَ لَسۡتُمۡ وَلَسۡتُمۡ بِاٰخِذِیۡہِ اِلَّاۤ اَنۡ تُغۡمِضُوۡا فِیۡہِ وَ اعۡلَمُوۡۤا وَاعۡلَمُوۡۤا اَنَّ اللّٰہَ غَنِیٌّ حَمِیۡدٌ اَلشَّیۡطٰنُ الشَّیۡطٰنُ یَعِدُکُمُ الۡفَقۡرَ وَ یَاۡمُرُکُمۡ وَیَاۡمُرُکُمۡ بِالۡفَحۡشَآءِ وَ اللّٰہُ وَاللّٰہُ یَعِدُکُمۡ مَّغۡفِرَۃً مِّنۡہُ وَ وَفَضۡلًا فَضۡلًا وَاللّٰہُ وَ اللّٰہُ وَاسِعٌ وٰسِعٌ عَلِیۡمٌ یُّؤۡتِی یُؤۡتِی الۡحِکۡمَۃَ مَنۡ یَّشَآءُ وَ مَنۡ وَمَنۡ یُّؤۡتَ الۡحِکۡمَۃَ فَقَدۡ اُوۡتِیَ خَیۡرًا کَثِیۡرًا وَ مَا وَمَا یَذَّکَّرُ اِلَّاۤ اُولُوا الۡاَلۡبَابِ الۡاَلۡبٰبِ وَ مَاۤ وَمَاۤ اَنۡفَقۡتُمۡ مِّنۡ نَّفَقَۃٍ اَوۡ نَذَرۡتُمۡ مِّنۡ نَّذۡرٍ فَاِنَّ اللّٰہَ یَعۡلَمُہٗ وَ مَا وَمَا لِلظّٰلِمِیۡنَ مِنۡ اَنۡصَارٍ اِنۡ تُبۡدُوا الصَّدَقٰتِ فَنِعِمَّا ہِیَ وَ اِنۡ وَاِنۡ تُخۡفُوۡہَا وَ تُؤۡتُوۡہَا وَتُؤۡتُوۡہَا الۡفُقَرَآءَ فَہُوَ خَیۡرٌ لَّکُمۡ وَ یُکَفِّرُ وَیُکَفِّرُ عَنۡکُمۡ مِّنۡ سَیِّاٰتِکُمۡ وَ اللّٰہُ وَاللّٰہُ بِمَا تَعۡمَلُوۡنَ خَبِیۡرٌ لَیۡسَ لَّیۡسَ عَلَیۡکَ ہُدٰٮہُمۡ وَ لٰکِنَّ وَلٰکِنَّ اللّٰہَ یَہۡدِیۡ مَنۡ یَّشَآءُ وَ مَا وَمَا تُنۡفِقُوۡا مِنۡ خَیۡرٍ فَلِاَنۡفُسِکُمۡ وَ مَا وَمَا تُنۡفِقُوۡنَ اِلَّا ابۡتِغَآءَ وَجۡہِ اللّٰہِ وَ مَا وَمَا تُنۡفِقُوۡا مِنۡ خَیۡرٍ الَّذِیۡ عَلَیۡہِ الۡحَقُّ سَفِیۡہًا اَوۡ ضَعِیۡفًا اَوۡ لَا یَسۡتَطِیۡعُ اَنۡ یُّمِلَّ ہُوَ فَلۡیُمۡلِلۡ وَلِیُّہٗ بِالۡعَدۡلِ وَ اسۡتَشۡہِدُوۡا وَاسۡتَشۡہِدُوۡا شَہِیۡدَیۡنِ مِنۡ رِّجَالِکُمۡ فَاِنۡ لَّمۡ یَکُوۡنَا رَجُلَیۡنِ فَرَجُلٌ وَّ امۡرَاَتٰنِ وَّامۡرَاَتَانِ مِمَّنۡ تَرۡضَوۡنَ مِنَ الشُّہَدَآءِ اَنۡ تَضِلَّ اِحۡدٰٮہُمَا فَتُذَکِّرَ اِحۡدٰٮہُمَا الۡاُخۡرٰی وَ لَا وَلَا یَاۡبَ الشُّہَدَآءُ اِذَا مَا دُعُوۡا وَ لَا وَلَا تَسۡـَٔمُوۡۤا اَنۡ تَکۡتُبُوۡہُ صَغِیۡرًا اَوۡ کَبِیۡرًا اِلٰۤی اَجَلِہٖ ذٰلِکُمۡ اَقۡسَطُ عِنۡدَ اللّٰہِ وَ وَاَقۡوَمُ اَقۡوَمُ لِلشَّہٰدَۃِ لِلشَّہَادَۃِ وَ اَدۡنٰۤی وَاَدۡنٰۤی اَلَّا تَرۡتَابُوۡۤا اِلَّاۤ اَنۡ تَکُوۡنَ تِجَارَۃً تِجٰرَۃً حَاضِرَۃً تُدِیۡرُوۡنَہَا بَیۡنَکُمۡ فَلَیۡسَ عَلَیۡکُمۡ جُنَاحٌ اَلَّا تَکۡتُبُوۡہَا وَ اَشۡہِدُوۡۤا وَاَشۡہِدُوۡۤا اِذَا تَبَایَعۡتُمۡ وَ لَا وَلَا یُضَآرَّ کَاتِبٌ وَّ لَا وَّلَا شَہِیۡدٌ وَ اِنۡ وَاِنۡ تَفۡعَلُوۡا فَاِنَّہٗ فُسُوۡقٌۢ بِکُمۡ وَ اتَّقُوا وَاتَّقُوا اللّٰہَ وَ یُعَلِّمُکُمُ اللّٰہُ وَ وَیُعَلِّمُکُمُ اللّٰہُ وَاللّٰہُ بِکُلِّ شَیۡءٍ عَلِیۡمٌ وَ اِنۡ وَاِنۡ کُنۡتُمۡ عَلٰی سَفَرٍ وَّ لَمۡ وَّلَمۡ تَجِدُوۡا کَاتِبًا فَرِہٰنٌ مَّقۡبُوۡضَۃٌ فَاِنۡ اَمِنَ بَعۡضُکُمۡ بَعۡضًا فَلۡیُؤَدِّ الَّذِی اؤۡتُمِنَ اَمَانَتَہٗ اَمٰنَتَہٗ وَ لۡیَتَّقِ وَلۡیَتَّقِ اللّٰہَ رَبَّہٗ وَ لَا وَلَا تَکۡتُمُوا الشَّہَادَۃَ الشَّہٰدَۃَ وَ مَنۡ وَمَنۡ یَّکۡتُمۡہَا فَاِنَّہٗۤ اٰثِمٌ قَلۡبُہٗ وَ اللّٰہُ وَاللّٰہُ بِمَا تَعۡمَلُوۡنَ عَلِیۡمٌ لِلّٰہِ لِّلّٰہِ مَا فِی السَّمٰوٰتِ وَ مَا وَمَا فِی الۡاَرۡضِ وَ اِنۡ وَاِنۡ تُبۡدُوۡا مَا فِیۡۤ اَنۡفُسِکُمۡ اَوۡ تُخۡفُوۡہُ یُحَاسِبۡکُمۡ بِہِ اللّٰہُ فَیَغۡفِرُ لِمَنۡ یَّشَآءُ وَ یُعَذِّبُ وَیُعَذِّبُ مَنۡ یَّشَآءُ وَ اللّٰہُ وَاللّٰہُ عَلٰی کُلِّ شَیۡءٍ قَدِیۡرٌ اٰمَنَ الرَّسُوۡلُ بِمَاۤ اُنۡزِلَ اِلَیۡہِ مِنۡ رَّبِّہٖ وَ الۡمُؤۡمِنُوۡنَ وَالۡمُؤۡمِنُوۡنَ کُلٌّ اٰمَنَ بِاللّٰہِ وَ وَمَلٰٓئِکَتِہٖ مَلٰٓئِکَتِہٖ وَکُتُبِہٖ وَ کُتُبِہٖ وَ رُسُلِہٖ وَرُسُلِہٖ لَا نُفَرِّقُ بَیۡنَ اَحَدٍ مِّنۡ رُّسُلِہٖ وَ قَالُوۡا وَقَالُوۡا سَمِعۡنَا وَ اَطَعۡنَا وَاَطَعۡنَا غُفۡرَانَکَ رَبَّنَا وَ اِلَیۡکَ وَاِلَیۡکَ الۡمَصِیۡرُ لَا یُکَلِّفُ اللّٰہُ نَفۡسًا اِلَّا وُسۡعَہَا لَہَا مَا کَسَبَتۡ وَ عَلَیۡہَا وَعَلَیۡہَا مَا اکۡتَسَبَتۡ رَبَّنَا لَا تُؤَاخِذۡنَاۤ اِنۡ نَّسِیۡنَاۤ اَوۡ اَخۡطَاۡنَا رَبَّنَا وَ لَا وَلَا تَحۡمِلۡ عَلَیۡنَاۤ اِصۡرًا کَمَا حَمَلۡتَہٗ عَلَی الَّذِیۡنَ مِنۡ قَبۡلِنَا رَبَّنَا وَ لَا وَلَا تُحَمِّلۡنَا مَا لَا طَاقَۃَ لَنَا بِہٖ وَ اعۡفُ وَاعۡفُ عَنَّا وَ اغۡفِرۡ وَاغۡفِرۡ لَنَا وَ ارۡحَمۡنَا وَارۡحَمۡنَاۤ اَنۡتَ مَوۡلٰٮنَا فَانۡصُرۡنَا عَلَی الۡقَوۡمِ الۡکٰفِرِیۡنَ بِسۡمِ اللّٰہِ الرَّحۡمٰنِ الرَّحِیۡمِ الٓمَّٓ الٓمّۤ اللّٰہُ لَاۤ اِلٰہَ اِلَّا ہُوَ الۡحَیُّ الۡقَیُّوۡمُ نَزَّلَ عَلَیۡکَ الۡکِتٰبَ بِالۡحَقِّ مُصَدِّقًا لِّمَا بَیۡنَ یَدَیۡہِ وَ اَنۡزَلَ وَاَنۡزَلَ التَّوۡرٰٮۃَ وَ الۡاِنۡجِیۡلَ وَالۡاِنۡجِیۡلَ مِنۡ قَبۡلُ ہُدًی لِّلنَّاسِ وَ اَنۡزَلَ وَاَنۡزَلَ الۡفُرۡقَانَ اِنَّ الَّذِیۡنَ کَفَرُوۡا بِاٰیٰتِ اللّٰہِ لَہُمۡ عَذَابٌ شَدِیۡدٌ وَ اللّٰہُ وَاللّٰہُ عَزِیۡزٌ ذُو انۡتِقَامٍ اِنَّ اللّٰہَ لَا یَخۡفٰی عَلَیۡہِ شَیۡءٌ فِی الۡاَرۡضِ وَ لَا وَلَا فِی السَّمَآءِ
